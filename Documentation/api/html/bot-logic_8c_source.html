00339 static void insert_behaviour_to_list(Behaviour_t * * list, Behaviour_t * behave) {
00340     Behaviour_t * ptr   = * list;
00341     Behaviour_t * temp;
00342 
00343     /* Kein Eintrag dabei? */
00344     if (behave == NULL) {
00345         return;
00346     }
00347 
00348     /* Erster Eintrag in der Liste? */
00349     if (ptr == NULL) {
00350         ptr = behave;
00351         *list = ptr;
00352     } else {
00353         /* Gleich mit erstem Eintrag tauschen? */
00354         if (ptr->priority < behave->priority) {
00355             behave->next = ptr;
00356             ptr = behave;
00357             *list = ptr;
00358         } else {
00359             /* Mit dem naechsten Eintrag vergleichen */
00360             while (NULL != ptr->next) {
00361                 if (ptr->next->priority < behave->priority) {
00362                     break;
00363                 }
00364 
00365                 /* Naechster Eintrag */
00366                 ptr = ptr->next;
00367             }
00368 
00369             temp = ptr->next;
00370             ptr->next = behave;
00371             behave->next = temp;
00372         }
00373     }

00447 static uint8_t isInCallHierarchy(Behaviour_t * job, Behaviour_t * caller_beh) {
00448     uint8_t level = 0;
00449 
00450     if (job == NULL) {
00451         LOG_DEBUG("kein Verhaltensdatensatz gegeben");
00452         return 0; // Liste ist leer
00453     }
00454 
00455     for (; job->caller; job = job->caller) {
00456         LOG_DEBUG("  ueberpruefe Verhalten %u mit caller %u...", job->priority, job->caller->priority);
00457         level++;
00458         if (job->caller == caller_beh) {
00459             LOG_DEBUG("   Verhalten %u wurde direkt von %u aufgerufen", job->priority, job->caller->priority);

00462     }
00463     return 0; // function kommt in Caller-Liste von job nicht vor

00471 void deactivate_called_behaviours(Behaviour_t * caller) {
00472     Behaviour_t * job; // Zeiger auf ein Verhalten
00473 
00474     LOG_DEBUG(""); // new line
00475     LOG_DEBUG("Callees von Verhalten %u sollen abgeschaltet werden.", get_behaviour(function)->priority);
00476     LOG_DEBUG("Beginne mit dem Durchsuchen der Liste");
00477     // Einmal durch die Liste gehen, und alle aktiven Funktionen pruefen, ob sie von dem uebergebenen Verhalten aktiviert wurden
00478     uint8_t i = 0;
00479     Behaviour_t * beh_of_function = NULL;
00480     for (job = behaviour; job; job = job->next) { // n mal
00481         if (job->active == BEHAVIOUR_ACTIVE) {
00482             i++;
00483             LOG_DEBUG("Verhalten mit Prio = %u ist ACTIVE, Durchlauf %u", job->priority, i);
00484             uint8_t level = isInCallHierarchy(job, caller); // O(n)
00485             LOG_DEBUG(" und hat Level %u Call-Abhaengigkeit", level);
00486             /* die komplette Caller-Liste (aber auch nur die) abschalten */
00487             Behaviour_t * ptr = job;
00488             for (; level > 0; --level) { // n mal
00489                 Behaviour_t * beh;
00490                 for (beh = behaviour; beh; beh = beh->next) { // n mal
00491                     /* Falls das Verhalten Caller eines anderen Verhaltens ist, duerfen wir es (noch) nicht deaktivieren! */
00492                     if (beh->caller == ptr) {
00493                         LOG_DEBUG("  Verhalten %u ist Caller eines anderen Verhaltens", ptr->priority);
00494                         break;
00495                     }
00496                 } // O(n)
00497                 Behaviour_t * tmp = ptr;
00498                 ptr = ptr->caller; // zur naechsten Ebene
00499                 if (beh == NULL) {
00500                     LOG_DEBUG("  Verhalten %u wird in Tiefe %u abgeschaltet", tmp->priority, level);
00501                     tmp->active = BEHAVIOUR_INACTIVE; // callee abschalten
00502                     tmp->caller = NULL; // Caller loeschen, damit Verhalten auch ohne BEHAVIOUR_OVERRIDE neu gestartet werden koennen
00503                 }
00504             }   // O(n^2)
00505             LOG_DEBUG(""); // new line
00506         }
00507         if (job == caller) {
00508             /* Verhalten fuer spaeter merken, wenn wir hier eh schon die ganze Liste absuchen */
00509             beh_of_function = job;
00510         }
00511     } // O(n^3)
00512     /* Verhaltenseintrag zu function benachrichtigen und wieder aktiv schalten */
00513     if (beh_of_function != NULL) {
00514         LOG_DEBUG("Verhalten %u wird aktiviert", beh_of_function->priority);
00515         beh_of_function->subResult = BEHAVIOUR_SUBCANCEL; // externer Abbruch
00516         beh_of_function->active = BEHAVIOUR_ACTIVE;
00517     }
00518 } // O(n^3)
