00151 static inline uint8_t lin_interpolate(uint8_t x1, uint8_t y1, uint8_t x2, uint8_t y2, uint8_t xs) {
00152     if (x1 == x2) {
00153         return 255;
00154     }
00155     uint16_t m = ((uint16_t) (y2 - y1) << 8) / (uint8_t) (x1 - x2); // m >= 0
00156     uint8_t x_diff = (uint8_t) (x1 - xs);
00157     return (uint8_t) ((uint8_t) ((x_diff * m) >> 8) + y1);  // m war kuenstlich um 8 Bit hochskaliert
00158 }
