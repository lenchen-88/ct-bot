00069 void bot_follow_line_behaviour(Behaviour_t * data) {
00070     switch (lineState) {
00071     case CHECK_LINE: /* sind beide Sensoren ueber einer Linie? */
00072         if (sensLineL>=LINE_SENSE&& sensLineR>=LINE_SENSE) {
00073             /* zunaechst alle Hilfsverhalten ausschalten, die den Algorithmus stoeren */
00074             /* Abgrund- und Kollisions-Verhalten ausschalten */
00075 #ifdef BEHAVIOUR_AVOID_COL_AVAILABLE
00076             deactivateBehaviour(bot_avoid_col_behaviour);
00077 #endif
00078 #ifdef BEHAVIOUR_AVOID_BORDER_AVAILABLE
00079             deactivateBehaviour(bot_avoid_border_behaviour);
00080 #endif
00081             /* bot_glance() stoert bot_turn() */
00082             //deactivateBehaviour(bot_glance_behaviour);
00083             /* losfahren und nach FOLLOW_LINE wechseln */
00084             speedWishLeft=BOT_SPEED_FOLLOW;
00085             speedWishRight=BOT_SPEED_FOLLOW;
00086             lineState=FOLLOW_LINE;
00087         }
00088         break;
00089 
00090     case FOLLOW_LINE:
00091         /* Pruefen, ob die Abgrundsensoren einen Abgrund sehen */
00092         if (sensBorderL>BORDER_DANGEROUS|| sensBorderR>BORDER_DANGEROUS) {
00093             /* Abgrund erkannt, das kann jetzt eine Linie sein oder ein richtiger Abgrund.*/
00094             if (sensBorderL>BORDER_DANGEROUS&& sensBorderR>BORDER_DANGEROUS) {
00095                 /* Wenn beidseitig erkannt, koennen wir damit nicht umgehen ->
00096                  * Ende des Verhaltens */
00097                 speedWishLeft=BOT_SPEED_STOP;
00098                 speedWishRight=BOT_SPEED_STOP;
00099                 //LOG_INFO("Stopp in FOLLOW_LINE");
00100                 return_from_behaviour(data);
00101                 break;
00102             }
00103             /* nachsehen, ob der linke oder rechte Liniensensor ohne Kontakt zur Linie ist
00104              * und ggfs. gegensteuern */
00105             if (sensBorderL>BORDER_DANGEROUS) {
00106                 cornerDetected=CORNER_LEFT;
00107             } else {
00108                 cornerDetected=CORNER_RIGHT;
00109             }
00110             /* nun zur vermuteten Ecke vorfahren */
00111             lineState=CHECK_BORDER;
00112             bot_drive_distance(data, 0, BOT_SPEED_FOLLOW, 3);
00113             break;
00114         }
00115         if (sensLineL<LINE_SENSE&& sensLineR>LINE_SENSE) {
00116             /* links von der Linie abgekommen, daher nach rechts drehen */
00117             //LOG_DEBUG("Drehe rechts");
00118             speedWishLeft=BOT_SPEED_FOLLOW;
00119             speedWishRight=-BOT_SPEED_FOLLOW;
00120         } else if (sensLineL>LINE_SENSE&& sensLineR<LINE_SENSE) {
00121             /* andersrum, also links drehen */
00122             //LOG_DEBUG("Drehe links");
00123             speedWishLeft=-BOT_SPEED_FOLLOW;
00124             speedWishRight=BOT_SPEED_FOLLOW;
00125         } else if (sensLineL>LINE_SENSE&& sensLineR>LINE_SENSE) {
00126             /* noch ueber der Linie, also einfach geradeaus weiter */
00127             //LOG_DEBUG("Fahre geradeaus");
00128             speedWishLeft=BOT_SPEED_FOLLOW;
00129             speedWishRight=BOT_SPEED_FOLLOW;
00130         }
00131         break;
00132 
00133     case CHECK_BORDER:
00134         /* wir sollten jetzt direkt an der Kante zum Abgrund stehen, wenn es
00135          * denn wirklich eine ist. In dem Fall fahren wir ein Stueck zurueck.
00136          * sonst gehen wir von einer Linie aus, drehen uns in die Richtung,
00137          * in der wir den "Abgrund" entdeckt haben und machen dann weiter mit
00138          * der Linienverfolgung */
00139         if (sensBorderL>BORDER_DANGEROUS|| sensBorderR>BORDER_DANGEROUS) {
00140             /* scheint wirklich ein Abgrund zu sein */
00141             lineState=RETREAT_AND_STOP;
00142             speedWishLeft=-BOT_SPEED_MAX;
00143             speedWishRight=-BOT_SPEED_MAX;
00144             break;
00145         }
00146         /* war nur eine Ecke, noch weiter darauf zu bis kein Kontakt mehr zur Linie */
00147         lineState=ADVANCE_CORNER;
00148         speedWishLeft=BOT_SPEED_FOLLOW;
00149         speedWishRight=BOT_SPEED_FOLLOW;
00150         break;
00151 
00152     case ADVANCE_CORNER:
00153         /* auf die Ecke zufahren, bis die Linie verschwindet */
00154         if (sensLineL<LINE_SENSE&& sensLineR<LINE_SENSE) {
00155             /* Linie weg, also Stop, kurz zurueck und drehen */
00156             lineState=CORNER_TURN;
00157             speedWishLeft=-BOT_SPEED_SLOW;
00158             speedWishRight=-BOT_SPEED_SLOW;
00159             break;
00160         }
00161         speedWishLeft=BOT_SPEED_FOLLOW;
00162         speedWishRight=BOT_SPEED_FOLLOW;
00163         break;
00164 
00165     case CORNER_TURN:
00166         /* 90 Grad in Richtung des detektierten Abgrunds drehen */
00167         lineState=CORRECT_POS;
00168         bot_turn(data, (cornerDetected==CORNER_LEFT) ? 90 : -90);
00169         cornerDetected=False;
00170         break;
00171 
00172     case CORRECT_POS:
00173         lineState=FOLLOW_LINE;
00174         bot_drive_distance(data, 0, BOT_SPEED_SLOW, 2);
00175         break;
00176 
00177     case RETREAT_AND_STOP:
00178         /* wir sind an einem Abgrund, Stop und Ende des Verhaltens */
00179         speedWishLeft=BOT_SPEED_STOP;
00180         speedWishRight=BOT_SPEED_STOP;
00181         return_from_behaviour(data);
00182         break;
00183     }
00184 }
