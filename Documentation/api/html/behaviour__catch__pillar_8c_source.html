00065     static uint8_t cancelcheck=False;
00066     static float angle;
00067 
00068     switch (catch_pillar_state){
00069         case START:
00070                startangle=heading_int;
00071                cancelcheck=False;
00072                catch_pillar_state=SEARCH_LEFT;
00073              break;
00074 
00075         case SEARCH_LEFT:
00076             // Nach 1x Rundumsuche und nix gefunden ist Schluss; Dazu wird der Check zum Start
00077             // nach Ueberschreiten der 5Grad-Toleranz eingeschaltet und die Drehung dann beendet wenn
00078             // der Winkel wieder innerhalb dieser Toleranz liegt
00079             if (cancelcheck) {
00080               if (fabs(heading - startangle) < 5 )
00081                 // Damit nicht endlos rundum gedreht wird Abbruch, wenn kein Gegenstand gefunden
00082                 catch_pillar_state = END;
00083             } else {
00084                 if (fabs(heading - startangle) >= 5)
00085                   cancelcheck=True;
00086             }
00087 
00088             // linker Sensor muss was sehen, der rechte aber nicht
00089             if (sensDistL < MAX_PILLAR_DISTANCE && sensDistR > MAX_PILLAR_DISTANCE){    // sieht der linke Sensor schon was?
00090                 angle=heading;
00091                 catch_pillar_state=LEFT_FOUND;
00092             } else
00093                 speedWishLeft=-BOT_SPEED_SLOW;
00094                 speedWishRight=+BOT_SPEED_SLOW;
00095                 //bot_turn(data,5); // Ein Stueck drehen
00096             break;
00097 
00098         case LEFT_FOUND:
00099             // links und rechts darf nicht gleichzeitig was gesehen werden, sonst weiter mit drehen
00100             if (sensDistL < MAX_PILLAR_DISTANCE && sensDistR < MAX_PILLAR_DISTANCE){
00101                 catch_pillar_state=SEARCH_LEFT;
00102                 break;
00103             }
00104 
00105             if (sensDistR < MAX_PILLAR_DISTANCE){   // sieht der rechte Sensor schon was?
00106                 angle= heading- angle;
00107                 if (angle < 0)
00108                     angle+=360;
00109                 angle= heading - angle/2;
00110                 catch_pillar_state=TURN_MIDDLE;
00111 //              bot_turn(data,-angle/2);
00112             } else
00113                 speedWishLeft=-BOT_SPEED_SLOW;
00114                 speedWishRight=+BOT_SPEED_SLOW;
00115 //              bot_turn(data,5);   // Eins Stueck drehen
00116             break;
00117 
00118         case TURN_MIDDLE:
00119                 if (fabs(heading - angle) > 2){
00120                     speedWishLeft=+BOT_SPEED_SLOW;
00121                     speedWishRight=-BOT_SPEED_SLOW;
00122                 } else {
00123                     bot_servo(data,SERVO1,DOOR_OPEN); // Klappe auf
00124                     catch_pillar_state=GO;
00125                 }

00127 
00128         case GO:

00130                 speedWishLeft=+BOT_SPEED_SLOW;
00131                 speedWishRight=+BOT_SPEED_SLOW;
00132                 // nicht endlos vorwaertslaufen, weiter mit Suche bis Drehende
00133                 if (sensDistL < MAX_PILLAR_DISTANCE && sensDistR < MAX_PILLAR_DISTANCE)
00134                     catch_pillar_state=SEARCH_LEFT;
00135             }
00136             else {
00137                   bot_servo(data,SERVO1,DOOR_CLOSE);
00138                   catch_pillar_state=END;

00140

00227 void bot_catch_pillar_behaviour(Behaviour_t * data) {
00228     static int16_t distLeft, distRight;
00229 
00230     switch (catch_pillar_state) {
00231     case START:
00232         bot_turn_maxspeed(data, 360, BOT_SPEED_SLOW);
00233         bot_cancel_behaviour(data, bot_turn_behaviour, turn_cancel_check);
00234         state_after_cancel = OBJECT_FOUND_AFTER_CANCEL;
00235         catch_pillar_state = END;
00236         break;
00237 
00238     case OBJECT_FOUND_AFTER_CANCEL: // Objekt erkannt
00239         // kommt hierher weil Cancel zugeschlagen hatte und Objekt erkannt wurde
00240         // ist schon zu weit gedreht, wieder etwas zurueckdrehen sonst genau Abstand messen
00241 
00242         // sehen beide Sensoren zu kleinen Abstand, dann ist dies wohl eine Wand und es geht weiter mit Drehen
00243         if (object_invalid()) {
00244             catch_pillar_state = OBJECT_INVALID;
00245             LOG_DEBUG("OBJECT_INVALID nach OBJECT_FOUND_AFTER_CANCEL");
00246             break;
00247         }
00248 
00249         // bei Gueltigkeit evtl. Korrektur notwendig
00250         catch_pillar_state = TURN_BACK;
00251         break;
00252 
00253     case TURN_BACK:
00254         BLOCK_BEHAVIOUR(data, 2000);    // Nachlauf abwarten
00255         if (turn_cancel_check()) {
00256             // Falls Sensor Objekt erkennt, genauen Abstand messen
00257             catch_pillar_state = MEASURE_DIST;
00258             break;
00259         }
00260         bot_turn_maxspeed(data, -20, BOT_SPEED_SLOW);
00261         bot_cancel_behaviour(data, bot_turn_behaviour, turn_cancel_check);
00262         state_after_cancel = OBJECT_FOUND_AFTER_CANCEL;
00263         catch_pillar_state = OBJECT_INVALID;
00264         break;
00265 
00266     case OBJECT_INVALID: {
00267         LOG_DEBUG("OBJECT_INVALID");
00268         int16_t turned = (int16_t)heading - start_heading;
00269         if (turned < 0) turned += 360;
00270         bot_turn_maxspeed(data, 360 - turned, BOT_SPEED_SLOW);
00271         bot_cancel_behaviour(data, bot_turn_behaviour, turn_cancel_check);
00272         state_after_cancel = OBJECT_FOUND_AFTER_CANCEL;
00273         catch_pillar_state = END;
00274         break;
00275     }
00276 
00277     case MEASURE_DIST:
00278         // genaue Entfernung zum Hindernis messen
00279         bot_measure_distance(data, &distLeft, &distRight, 15);
00280         catch_pillar_state = OPEN_DOOR;
00281         break;
00282 
00283     case OPEN_DOOR:
00284         if (distLeft > MAX_PILLAR_DISTANCE && distRight > MAX_PILLAR_DISTANCE) {
00285             // bei ungueltigem Abstand nach Messung Restdrehung
00286             catch_pillar_state = OBJECT_INVALID;
00287             LOG_DEBUG("OBJECT_INVALID nach OPEN_DOOR");
00288             break;
00289         }
00290         catch_pillar_state = OBJECT_FOUND_AFTER_MEASURE;
00291         break;
00292 
00293     case OBJECT_FOUND_AFTER_MEASURE:
00294         // nehme den Sensorabstand vom Sensor, welcher auch Objekt sieht
00295         if (distLeft <= MAX_PILLAR_DISTANCE) {
00296             obj_posL = calc_point_in_distance(heading, DISTSENSOR_POS_FW + distLeft, DISTSENSOR_POS_SW);
00297             catch_pillar_state = TURN_TO_RIGHT_SENSOR;
00298         } else {
00299             obj_posR = calc_point_in_distance(heading, DISTSENSOR_POS_FW + distRight, -DISTSENSOR_POS_SW);
00300             catch_pillar_state = GO_TO_POINT;
00301             bot_servo(data, SERVO1, DOOR_OPEN); // Klappe auf
00302         }
00303         break;
00304 
00305     case TURN_TO_RIGHT_SENSOR:
00306         check_sensorside = 1;
00307         catch_pillar_state = START;
00308         break;
00309 
00310     case GO_TO_POINT:
00311         bot_goto_pos(data, ((obj_posL.x + obj_posR.x) / 2), ((obj_posL.y
00312                 + obj_posR.y) / 2), 999);
00313         catch_pillar_state = CLOSE_DOOR;
00314         break;
00315 
00316     case CLOSE_DOOR:
00317         if (sensTrans == 1) {
00318             // Klappe schliessen falls Objekt eingefangen wurde
00319             bot_servo(data, SERVO1, DOOR_CLOSE);
00320         }
00321         catch_pillar_state = END;
00322         break;
00323 
00324     default:
00325         exit_behaviour(data, sensTrans); // == BEHAVIOUR_SUBSUCCESS, falls Objekt eingefangen
00326         break;
00327     }

00404 void bot_catch_pillar_behaviour(Behaviour_t * data) {
00405     static position_t obj_pos;
00406 
00407     switch (catch_pillar_state) {
00408     /* Auf los geht's los */
00409     case START:
00410         /* Drehen mit Abbruch bei Objekterkennung */
00411         bot_turn_maxspeed(data, max_turn, BOT_SPEED_SLOW);
00412         bot_cancel_behaviour(data, bot_turn_behaviour, turn_cancel_check);
00413         side = 0;
00414         headingL = -1;
00415         headingR = -1;
00416         state_after_cancel = OBJECT_FOUND;
00417         catch_pillar_state = END;
00418         break;
00419 
00420         /* Objekt erkannt */
00421     case OBJECT_FOUND:
00422         if (headingL < 0 || headingR < 0) {
00423             /* Erkennung fehlgeschlagen */
00424             catch_pillar_state = END;
00425             return;
00426         }
00427         catch_pillar_state = OPEN_DOOR;
00428         /* Abstand zum Objekt berechnen aus gedrehtem Winkel */
00429         int16_t dHead = headingR - headingL;
00430         if (dHead < 0) {
00431             dHead += 360;
00432         }
00433         int16_t dist = (int16_t) ((DISTSENSOR_POS_SW * 2.0f) / dHead * (180.0f / M_PI));
00434         /* Objektkoordis berechnen */
00435         obj_pos = calc_point_in_distance(headingR, /*DISTSENSOR_POS_FW + */ dist,
00436                 -DISTSENSOR_POS_SW);
00437         break;
00438 
00439         /* Klappe auf */
00440     case OPEN_DOOR:
00441         catch_pillar_state = GO_TO_POINT;
00442         bot_servo(data, SERVO1, DOOR_OPEN);
00443         break;
00444 
00445         /* zum Objekt fahren und Stopp, wenn das Objekt im Fach erkannt wird */
00446     case GO_TO_POINT:
00447         bot_goto_pos(data, obj_pos.x, obj_pos.y, 999);
00448         bot_cancel_behaviour(data, bot_goto_pos_behaviour, goto_pos_cancel);
00449         catch_pillar_state = CLOSE_DOOR;
00450         break;
00451 
00452         /* Klappe zu */
00453     case CLOSE_DOOR:
00454         if (sensTrans == 1) {
00455             // Klappe schliessen falls Objekt eingefangen wurde
00456             bot_servo(data, SERVO1, DOOR_CLOSE);
00457         }
00458         catch_pillar_state = END;
00459         break;
00460 
00461         /* Ende */
00462     default:
00463         exit_behaviour(data, sensTrans); // == BEHAVIOUR_SUBSUCCESS, falls Objekt eingefangen
00464         break;
00465     }
