00027 /* Das bot_follow_object-Verhalten laesst den Bot ein (bewegliches) Objekt verfolgen. Kommt umgekehrt das
00028  * Objekt dichter als 20 cm an den Bot heran, "flieht" er vor ihm und faehrt rueckwaerts.
00029  * Die Ausrichtung der Bot-Front ist immer (moeglichst) parallel zur Front des anderen
00030  * Objekts. Ist dieses z.B. ein Fahrzeug, dann faehrt der Bot durch diese Winkelanpassung
00031  * dem anderen Fahrzeug hinterher (oder rueckwaerts vor ihm weg).
00032  * Das Objekt muss breiter sein, als der Abstand der beiden Distanzsensoren, damit es
00033  * korrekt erkannt wird.
00034  * Sieht der Bot kein Objekt im Erfassungsbereich der Distanzsensoren, bleibt er stehen.
00035  *
00036  * Das Verhalten bietet noch sehr viel Spielraum fuer Anpassungen und eigene Experimente!
00037  * Insbesondere ist es bisher nicht sehr robust gegenueber grossen Entfernungsaenderungen
00038  * in kurzer Zeit. Die besten Ergebnisse werden erzielt, wenn sich das andere Objekt
00039  * langsam und kontinuierlich bewegt.
00040  * Obwohl der Verfolgungsalgorithmus ziemlich simpel gestrickt ist, bekommt der Kleine mit
00041  * ihm eine Art "Eigenleben" und es ist recht unterhaltsam, damit zu spielen ;-)
00042  * Sinnvolle Erweiterungen waeren ein Suchalgorithmus, falls kein Objekt im Erfassungsbereich
00043  * ist, oder ein Filter fuer die Werte der Distanzsensoren.
00044  */

00064 static const uint8_t FOLLOW_DISTANCE    = 200;  
00065 static const uint8_t MAX_SPEED          = 200;  
00073 static void update_speed(int16_t * pSensDist, int16_t * pSpeed) {
00074     /* Entfernung zu gross -> hinterher! */
00075     if (*pSensDist > FOLLOW_DISTANCE) *pSpeed += (*pSensDist - FOLLOW_DISTANCE) / (FOLLOW_DISTANCE / 10);
00076 
00077     /* Entfernung zu klein -> bitte Abstand halten! */
00078     if (*pSensDist < (FOLLOW_DISTANCE - 20)) *pSpeed -= ((int16_t) (FOLLOW_DISTANCE * 1.5f) - *pSensDist)
00079         / ((int16_t) (FOLLOW_DISTANCE * 1.5f) / 10);
00080 
00081     /* Objekt wid nur noch halb erfasst -> abbiegen! */
00082     if (sensDistL - *pSensDist > FOLLOW_DISTANCE || sensDistR - *pSensDist > FOLLOW_DISTANCE) {
00083         // Drehungen teilweise zu stark => besser neu suchen in diesem Fall
00084 //      *pSpeed -= 1;
00085     }
00086 
00087     /* anti wind-up */
00088     if (*pSpeed > MAX_SPEED) *pSpeed = MAX_SPEED;
00089     else if (*pSpeed < -MAX_SPEED) *pSpeed = -MAX_SPEED;
00090 
00091     /* Abstand ok -> Pause einlegen */
00092     if (abs(*pSensDist - FOLLOW_DISTANCE) <= 10) *pSpeed = BOT_SPEED_STOP;
00093 }

00099 void bot_follow_object_behaviour(Behaviour_t * data) {
00100     static uint8_t distToggle = 0;
00101 #ifdef BEHAVIOUR_DRIVE_STACK_AVAILABLE
00102     static uint8_t movedState = 0;
00103 #endif
00104     switch (state) {
00105     case 0:
00106         /* auf neue Messwerte der Distanzsensoren warten */
00107         if (distToggle != sensDistLToggle) {
00108             distToggle = sensDistLToggle;
00109             state = 1;
00110         }
00111         break;
00112 
00113     case 1:
00114         /* go! */
00115         update_speed(&sensDistL, &speedLeft);   // links
00116         update_speed(&sensDistR, &speedRight);  // rechts
00117 
00118         /* ganz allein :( -> anhalten */
00119         if (sensDistL == SENS_IR_INFINITE && sensDistR == SENS_IR_INFINITE) {
00120 #ifdef BEHAVIOUR_DRIVE_STACK_AVAILABLE
00121             /* zurueck zum Startpunkt mit bot_drive_stack(), falls
00122              * mindestens 8 Sekunden kein Objekt in Sichtweite */
00123             if (movedState == 0) {
00124                 lastMove = TIMER_GET_TICKCOUNT_32;
00125                 movedState = 1;
00126             } else {
00127                 if (timer_ms_passed_32(&lastMove, 10000)) {
00128                     bot_drive_stack(data);
00129                     movedState = 0;
00130                 }
00131             }
00132 #endif  // BEHAVIOUR_DRIVE_STACK_AVAILABLE
00133 
00134             speedLeft   = BOT_SPEED_STOP;
00135             speedRight  = BOT_SPEED_STOP;
00136             break;
00137         }
00138         state = 0;
00139         break;
00140 
00141     case 2:
00142         /* hier koennte man abbrechen, wenn man denn wollte...
00143          * Eigentlich gibt es aber keinen Zeitpunkt, an dem
00144          * sich das Verhalten selbst beenden sollte.
00145          */
00146 //      return_from_behaviour(data);
00147         break;
00148     }
00149 
00150     /* Geschwindigkeiten setzen */
00151     speedWishLeft   = speedLeft;
00152     speedWishRight  = speedRight;
00153 }
