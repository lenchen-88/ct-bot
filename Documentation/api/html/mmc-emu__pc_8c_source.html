00027 /* Die PC-Emulation einer MMC / SD-Card ermoeglicht es, am PC dieselben Funktionen zu benutzen, wie bei einer echten
00028  * MMC / SD-Card. Als Speichermedium dient hier eine Datei (MMC_EMU_FILE), deren Groesse sich mit MMC_EMU_SIZE in Byte 
00029  * einstellen laesst. Ist die Datei nicht vorhanden oder derzeit kleiner als MMC_EMU_SIZE, wird sie angelegt bzw. 
00030  * vergroessert. Achtung, startet man den C-Bot vom Sim aus, liegt die Datei, falls kein absoluter Pfad angegeben wurde, im
00031  * Verzeichnis dem ct-Sims, von der Konsole aus gestartet erwartet / erzeugt der Code die Datei im Unterverzeichnis 
00032  * "Debug-Linux" bzw. "Debug-W32".
00033  * Eine sinnvolle (und die derzeit einzig moegliche) Verwendung der Emulation ergibt sich im Zusammenspiel mit dem
00034  * Virtual Memory Management fuer MMC. Ein Verhalten kann in diesem Fall immer auf dieselbe Art und Weise Speicher anfordern,
00035  * je nach System liegt dieser physisch dann entweder auf einer MMC / SD-Card (MCU) oder in einer Datei (PC). Fuer das
00036  * Verhalten ergibt sich kein Unterschied und es kann einfach derselbe Code verwendet werden.
00037  * Moechte man die Funktion mmc_fopen() benutzen, also auf FAT16-Dateien zugreifen, so ist zu beachten, dass deren "Dateiname"
00038  * in der Datei fuer die Emulation am Anfang eines 512 Byte grossen Blocks steht (denn auf einer echten MMC / SD-Card ezeugt
00039  * ein Betriebssystem eine neue Datei immer am Anfang eines Clusters und mmc_fopen() sucht nur dort nach dem "Dateinamen").
00040  * Im Moment gibt es noch keine Funktion zum Anlegen einer neuen Datei auf einer echten oder emulierten MMC / SD-Card. 
00041  * Die Code der Emulation ist voellig symmetrisch zum Code fuer eine echte MMC / SD-Card aufgebaut.  
00042  */

00098 uint8_t mmc_emu_init(void) {

00100     mmc_emu_file = fopen(MMC_EMU_FILE, "r+b");      // Datei versuchen zu oeffnen

00102         /* Datei existiert noch nicht oder kann nicht erzeugt werden */

00104         if (mmc_emu_file == NULL) {
00105             /* Datei kann nicht erzeugt werden */
00106             mmc_emu_init_state = 1;
00107             return 1;

00109     }
00110     if (mmc_emu_get_size() < MMC_EMU_SIZE) {
00111         /* vorhandene Datei ist zu klein, also auf MMC_EMU_SIZE vergroessern */
00112         mmc_emu_init_state = 1;

00114         if (putc(0, mmc_emu_file) != 0) return 3;

00116         mmc_emu_init_state = 0;
00117     }
