00118 static inline float calc_weighting(void) {
00119     if (encoderRateInfo[0] <= 6 || encoderRateInfo[1] <= 6) {
00120         /* mindestens ein Rad steht still => ganz schlecht */
00121         vm_left = FLT_MAX/2;
00122         vm_right = FLT_MAX/2;
00123         return FLT_MAX/2;
00124     }
00125 
00126     /* Mittelwert und Varianz links und rechts aktualisieren */
00127     xm_left = xm_left * (1.0f - a) + (float)encoderRateInfo[0] * a;
00128     xm_right = xm_right * (1.0f - a) + (float)encoderRateInfo[1] * a;
00129     vm_left = vm_left * (1.0f - a) + pow((float)encoderRateInfo[0] - xm_left, 2) * a;
00130     vm_right = vm_right * (1.0f - a) + pow((float)encoderRateInfo[1] - xm_right, 2) * a;
00131 
00132     /* Durchschnitt von Varianz links und rechts ergibt neue Bewertung */
00133     return vm_left * 0.5 + vm_right * 0.5;
00134 }

00156 static void compare_weightings(const uint16_t dt, int8_t * pid_param, const int8_t step) {
00157     /* Mittelwerte und Varianzen aktualisieren */
00158     float weight = calc_weighting();
00159 
00160     /* nach dt ms aktuelle Bewertung mit bisher bester vergleichen */
00161     if (timer_ms_passed_32(&ticks, dt)) {
00162         uint32_t dt_real = TIMER_GET_TICKCOUNT_32 - ticks_start;
00163         weight = weight / (float)dt_real * (dt*1000.0f/176.0f);     // Bewertung normieren
00164         if (best_weight >= weight) {
00165             /* Verbesserung => die aktuellen Parameter merken */
00166             best_weight = weight;
00167             best_Kp = Kp;
00168             best_Ki = Ki;
00169             best_Kd = Kd;
00170 
00171             /* User per LOG informieren */
00172             LOG_INFO("Kp=%d\tKi=%d\tKd=%d\tnach %u s", best_Kp, best_Ki, best_Kd, TICKS_TO_MS(TIMER_GET_TICKCOUNT_32)/1000);
00173 //          uint32_t weight_int = weight * 1000000.0f;
00174             LOG_DEBUG("weight=%lu", (uint32_t)(weight * 1000000.0f));
00175         }
00176 
00177         /* neuen Parameter einstellen und Nachlauf abwarten */
00178         *pid_param = (int8_t) (*pid_param + step);
00179

00183         wait_for_stop();

00199 static void find_Kp_region(void) {
00200     /* Go! */
00201     speedWishLeft = -cal_speed;
00202     speedWishRight = cal_speed;
00203 
00204     /* Kp alle 15 s um 10 erhoehen und besten Parameter speichern */
00205     compare_weightings(15000, &Kp, 10);
00206 
00207     if (Kp < 0 || Kp >= max_Kp) {
00208         /* alle Kp ueberprueft => nun Ki suchen */

00210 
00211         /* wir suchen Ki in der Umgebung (+/- 10) von best_Kp */
00212         Kp = (int8_t) (best_Kp > 10 ? best_Kp - 10 : 0);
00213         Kp_region_end = (int8_t) (best_Kp < 116 ? best_Kp + 10 : 120);
00214         Ki = 1;
00215         best_weight = FLT_MAX;
