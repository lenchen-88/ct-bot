00091 /*
00092  * Eine Karte ist wie folgt organisiert:
00093  * Es gibt Sektionen zu je MAP_SECTION_POINTS * MAP_SECTION_POINTS.
00094  * Diese Sektionen enthalten direkt die Pixel-Daten
00095  * Auf dem PC liegen die Daten genauso im RAM, wie sie auf der MMC
00096  * angeordnet sind.
00097  *
00098  * Auf dem MCU passt eine Sektion in einen Flashblock, bzw immer 2
00099  * Sektionen in einen Block Es stehen immer 2 Sections also 1
00100  * Flash-Block im SRAM und werden bei Bedarf gewechselt.
00101  *
00102  * Felder sind vom Typ int8_t und haben einen Wertebereich von -128
00103  * bis 127.
00104  * 0 bedeutet: wir wissen nichts ueber das Feld
00105  * negative Werte bedeuten: Hindernis
00106  * positive Werte bedeuten: freier Weg
00107  * Je groesser der Betrag ist, desto sicherer die Aussage ueber das
00108  * Feld.
00109  * Der Wert -128 ist Loechern vorbehalten und wird dann auch nicht
00110  * durch die Abstandssensoren veraendert.
00111  *
00112  * Felder werden wie folgt aktualisiert:
00113  * Wird ein Punkt als frei betrachtet, erhoehen wir den Wert des
00114  * Feldes um MAP_STEP_FREE.
00115  * Wird ein Punkt als belegt erkannt, ziehen wir um ihn einen
00116  * Streukreis mit dem Radius MAP_RADIUS.
00117  * Das Zentrum des Kreises wird um MAP_STEP_OCCUPIED dekrementiert,
00118  * nach aussen hin immer weniger.
00119  * Wird ein Feld als Loch erkannt, setzen wir den Wert fest auf -128.
00120  */

00485 static map_section_t * get_section(int16_t x, int16_t y) {
00486     // Da immer 2 Sections in einem Block stehen: richtige der beiden Sections raussuchen
00487     uint8_t index = (uint8_t)((x / MAP_SECTION_POINTS) & 0x01);
00488 
00489     /* Sicherheitscheck */
00490     if (((uint16_t) x >= (MAP_SIZE * MAP_RESOLUTION)) || ((uint16_t) y >= (MAP_SIZE * MAP_RESOLUTION))) {
00491         LOG_ERROR("Versuch auf ein Feld ausserhalb der Karte zu zugreifen!! x=%u y=%u", x, y);
00492         return map[0];
00493     }
00494 
00495     /* Berechne den gesuchten Block */
00496     uint16_t local_x = (uint16_t) x % MACRO_BLOCK_LENGTH;
00497     // wenn MACRO_BLOCK_LENGTH eine 2er Potenz ist, optimiert der Compiler hier
00498     uint16_t local_y = (uint16_t) y % MACRO_BLOCK_LENGTH;
00499 
00500 #ifdef DEBUG_STORAGE
00501     LOG_DEBUG("local_x= %d, local_y= %d", local_x, local_y);
00502 #endif
00503 
00504     uint16_t block = local_x / MAP_SECTION_POINTS + (local_y / MAP_SECTION_POINTS)
00505             * (MACRO_BLOCK_LENGTH / MAP_SECTION_POINTS);
00506 
00507     block = block >> 1; // es passen immer 2 Sections in einen Block
00508 
00509     /* Makroblock berechnen */
00510     uint16_t macroblock = (uint16_t) x / MACRO_BLOCK_LENGTH + ((uint16_t) y / MACRO_BLOCK_LENGTH) * MAP_LENGTH_IN_MACRO_BLOCKS;
00511 
00512 #ifdef DEBUG_STORAGE
00513     LOG_DEBUG("Macroblock= %u", macroblock);
00514 #endif
00515 
00516     block += macroblock * MACRO_BLOCK_LENGTH * (MACRO_BLOCK_LENGTH / 512); // noch in den richtigen Makroblock springen
00517 
00518 #ifdef DEBUG_STORAGE
00519     LOG_DEBUG("block= %u", block);
00520 #endif
00521 
00522     /* Ist der Block schon geladen? */
00523     if (map_current_block.block == block) {
00524 #ifdef DEBUG_STORAGE
00525         LOG_DEBUG("ist noch im Puffer");
00526 #endif
00527         return map[index];
00528     }
00529 
00530     /* Block ist also nicht im Puffer */
00531 #ifdef DEBUG_STORAGE
00532     LOG_DEBUG("ist nicht im Puffer");
00533 #endif
00534 
00535     /* Wurde der Block im RAM veraendert? */
00536     if (map_current_block.updated == True) {
00537         /* Shrinking */
00538         if (map_current_block.x < map_min_x) {
00539             map_min_x = map_current_block.x;
00540 #ifdef BOT_FS_AVAILABLE
00541             min_max_updated = True;

00758 static void update_field_circle(int16_t x, int16_t y, int8_t radius, int8_t value) {

00760

00773             starty = sec_y * MAP_SECTION_POINTS;
