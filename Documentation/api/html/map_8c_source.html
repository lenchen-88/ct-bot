00089 /*
00090  * Eine Karte ist wie folgt organisiert:
00091  * Es gibt Sektionen zu je MAP_SECTION_POINTS * MAP_SECTION_POINTS.
00092  * Diese Sektionen enthalten direkt die Pixel-Daten
00093  * Auf dem PC liegen die Daten genauso im RAM, wie sie auf der MMC
00094  * angeordnet sind.
00095  *
00096  * Auf dem MCU passt eine Sektion in einen Flashblock, bzw immer 2
00097  * Sektionen in einen Block Es stehen immer 2 Sections also 1
00098  * Flash-Block im SRAM und werden bei Bedarf gewechselt.
00099  *
00100  * Felder sind vom Typ int8_t und haben einen Wertebereich von -128
00101  * bis 127.
00102  * 0 bedeutet: wir wissen nichts ueber das Feld
00103  * negative Werte bedeuten: Hindernis
00104  * positive Werte bedeuten: freier Weg
00105  * Je groesser der Betrag ist, desto sicherer die Aussage ueber das
00106  * Feld.
00107  * Der Wert -128 ist Loechern vorbehalten und wird dann auch nicht
00108  * durch die Abstandssensoren veraendert.
00109  *
00110  * Felder werden wie folgt aktualisiert:
00111  * Wird ein Punkt als frei betrachtet, erhoehen wir den Wert des
00112  * Feldes um MAP_STEP_FREE.
00113  * Wird ein Punkt als belegt erkannt, ziehen wir um ihn einen
00114  * Streukreis mit dem Radius MAP_RADIUS.
00115  * Das Zentrum des Kreises wird um MAP_STEP_OCCUPIED dekrementiert,
00116  * nach aussen hin immer weniger.
00117  * Wird ein Feld als Loch erkannt, setzen wir den Wert fest auf -128.
00118  */

00500 #endif

00748 static void update_field_circle(int16_t x, int16_t y, int8_t radius, int8_t value) {

00750 
00751     int16_t sec_x_max = (x + radius) / MAP_SECTION_POINTS;
00752     int16_t sec_x_min = (x - radius) / MAP_SECTION_POINTS;
00753     int16_t sec_y_max = (y + radius) / MAP_SECTION_POINTS;
00754     int16_t sec_y_min = (y - radius) / MAP_SECTION_POINTS;

00757     int16_t sec_x, sec_y, X, Y, dX, dY;
00758     int16_t starty, startx, stopx, stopy;

00760     for (sec_y = sec_y_min; sec_y <= sec_y_max; sec_y++) {
00761         // Bereich weiter eingrenzen
00762         if (sec_y * MAP_SECTION_POINTS > (y - radius)) {
00763             starty = sec_y * MAP_SECTION_POINTS;
00764         } else {
00765             starty = y - radius;
00766         }
00767         if ((sec_y + 1) * MAP_SECTION_POINTS < (y + radius)) {
00768             stopy = (sec_y + 1) * MAP_SECTION_POINTS;
00769         } else {
00770             stopy = y + radius;
00771         }
00772 
00773         for (sec_x = sec_x_min; sec_x <= sec_x_max; sec_x++) {

00775                 startx = sec_x * MAP_SECTION_POINTS;

00777                 startx = x - radius;

00780                 stopx = (sec_x + 1) * MAP_SECTION_POINTS;

00782                 stopx = x + radius;
