00063 void bot_goto_obstacle_behaviour(Behaviour_t * data) {
00064     static int16_t distLeft, distRight;
00065     switch (obst_state) {
00066     case 0:
00067         /* Entfernung zum Hindernis messen */
00068         bot_measure_distance(data, &distLeft, &distRight, 15);
00069         LOG_DEBUG("Hindernis ist %d|%d mm entfernt", distLeft, distRight);
00070         obst_state = 1;
00071         break;
00072     case 1: {
00073         /* kleinere Distanz als Richtwert benutzen */
00074         int16_t dist = distLeft < distRight ? distLeft : distRight;
00075         if (dist <= 600) {
00076             /* Entfernung - gewuenschte Entfernung fahren */
00077             int16_t to_drive = dist - obst_distance;
00078             LOG_DEBUG("to_drive=%d", to_drive);
00079             if (abs(to_drive) > TARGET_MARGIN) {
00080                 bot_goto_dist(data, abs(to_drive), sign16(to_drive));
00081                 obst_state = 0;
00082             } else {
00083                 obst_state = 2;
00084             }
00085         } else {
00086             /* kein Hindernis in Sichtweite, also erstmal vorfahren */
00087             LOG_DEBUG("Noch kein Hindernis in Sichtweite");
00088             bot_goto_dist(data, 400, 1);
00089             obst_state = 0;
00090         }
00091         break;
00092     }
00093     case 2:
00094         if (obst_parallel == 1) {
00095             /* heading korrigieren, so dass Bot parallel zum Hindernis steht */
00096             int16_t diff = distRight - distLeft;
00097             if (abs(diff) < max_par_diff) {
00098                 /* unsinnig bei zu grosser Differenz */
00099                 LOG_DEBUG("diff=%d", diff);
00100                 float alpha = (float)diff / (float)(DISTSENSOR_POS_SW*2);
00101                 alpha /= 2.0*M_PI/360.0;
00102                 LOG_DEBUG("alpha=%f", alpha);
00103                 bot_turn(data, (int16_t)alpha);
00104             }
00105         }
00106         obst_state = 3;
00107         break;
00108     case 3:
00109         /* fertig :-) */
00110         return_from_behaviour(data);
00111         break;
00112     }
00113 }
