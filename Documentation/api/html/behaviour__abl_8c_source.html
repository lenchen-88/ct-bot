00308     /* check for comment */
00309     if (abl_i_cache[0] == '/' || abl_i_cache[0] == '\0') {
00310         LOG_DEBUG("comment found: %s", abl_i_cache);

00312     }
00313     /* search end of name / keyword */
00314     char * func_end = strchr(abl_i_cache, '(');
00315 #ifdef ERROR_CHECKS
00316     if (func_end == NULL) {
00317         LOG_ERROR("syntax error, instruction was:");
00318         LOG_ERROR("%s", abl_i_cache);
00319         return I_UNKNOWN;
00320     }
00321 #endif // ERROR_CHECKS
00322     *func_end = '\0';
00323     ++func_end;
00324     /* find and parse all parameters between '(' and ')' */
00325     char * par_end = func_end; // start with first argument-char
00326     uint8_t i;
00327     for (i = 0; *par_end != ')'; ++i) {
00328         par_end = strchr(func_end, ',');
00329         if (par_end == NULL) {
00330             par_end = strchr(func_end, ')');
00331         }
00332 #ifdef ERROR_CHECKS
00333         if (par_end == NULL) {
00334             LOG_ERROR("syntax error, instruction was:");
00335             LOG_ERROR("%s", abl_i_cache);
00336             return I_UNKNOWN;
00337         }
00338 #endif // ERROR_CHECKS
00339         if (func_end != par_end) {
00340             /* convert parameter */
00341             LOG_DEBUG("%u. parameter found, trying to parse...", i + 1);
00342             abl_params[i] = parameter_parse(func_end, par_end - 1);
00343             func_end = par_end + 1;
00344         }
00345     }
00346     /* find meaning of actual keyword */
00347     uint8_t j;
00348     for (j = 0; j < sizeof(keywords) / sizeof(PGM_P); ++j) {
00349         memcpy_P(&p_keywords, &keywords[j], sizeof(PGM_P));
00350         if (strcmp_P(abl_i_cache, p_keywords) == 0) {
00351 #ifdef PC
00352             LOG_DEBUG("%s decoded", keywords[j]);
00353 #endif
00354             return j;
00355         }
00356     }
00357     /* no keyword found, so instruction must be a function call */
00358     LOG_DEBUG("function name decoded as \"%s\"", abl_i_cache);

00360     case 0:
00361         LOG_DEBUG("function call decoded: \"%s()\"", abl_i_cache);

00380 static instruction_t i_fetch(void) {
00381     LOG_DEBUG("i_fetch() entered");
00382     /* check for valid data */
00383     if (p_abl_i_data == NULL) {
00384         ip = abl_prg_data;
00385         return I_UNKNOWN;
00386     }
00387     /* update ip, skip spaces and tabs */
00388     do {
00389         ip++;
00390     } while (ip <= &p_abl_i_data[510] /*&& ip >= p_abl_i_data*/ && (*ip == 0x20 || *ip == 0x9));
00391     LOG_DEBUG("ip=0x%04x i_start=0x%04x", ip - p_abl_i_data, i_start - abl_i_cache);
00392     /* ip becomes start of this instruction */
00393     const char * old_ip = ip;
00394     uint8_t len;
00395     uint8_t max_len = (uint8_t) ((ABL_INSTRUCTION_LENGTH + 1) - (size_t) i_start + (size_t) abl_i_cache);
00396     /* find end of this instruction (=end of line) */
00397     for (len = 0; len < max_len; ++len, ++ip) {
00398         if (*ip == '\n') {
00399             /* found, instruction complete */
00400             break;
00401         }
00402         /* --> monsters here - don't touch this code <-- */
00403             else if (ip >= &p_abl_i_data[511] || ip <= p_abl_i_data - 1) {
00404                 /* actual instruction is not completely in this buffer :( */
00405                 LOG_DEBUG("instruction overlaps code segment, loading next one...");
00406                 /* save the part we already found */
00407                 len = (uint8_t) (ip - old_ip + 1);
00408                 /* skip left spaces and tabs in old segement */
00409                 while (len > 0 && (*old_ip == 0x20 || *old_ip == 0x9)) {
00410                     ++old_ip;
00411                     --len;
00412                 }
00413                 memcpy(i_start, old_ip, len);
00414                 i_start[len] = '\0';
00415                 LOG_DEBUG("first i-part is \"%s\"", i_start);
00416                 /* swap-in new program-segment and adjust pointers */
00417                 load_program(1);
00418                 ip = p_abl_i_data;
00419                 --ip;
00420                 i_start = &abl_i_cache[len]; // don't overwrite already copied part of instruction!
00421                 return i_fetch(); // complete instruction recursively
00422             }
00423         else if (*ip == '\0') {
00424             /* end of program reached! */
00425             return I_UNKNOWN;
00426         }
00427     }
00428     /* check length of found instruction */
00429     if (len >= max_len - 1) {
00430 #ifdef ERROR_CHECKS
00431         LOG_ERROR("%s", old_ip);
00432         LOG_ERROR("instruction is longer than %u bytes, increase ABL_INSTRUCTION_LENGTH!", ABL_INSTRUCTION_LENGTH);
00433 #endif
00434         return I_UNKNOWN;
00435     }
00436     /* copy instruction into i-cache */
00437     memcpy(i_start, old_ip, len);
00438     i_start[len] = '\0';
00439     LOG_DEBUG("fetched instruction \"%s\" is %u bytes long", abl_i_cache, &i_start[len] - abl_i_cache);
00440     i_start = abl_i_cache; // correct start of i-cache (if we came here recursively)
00441     /* decode this instruction */
00442     return i_decode();
00443 }

00480     /* create new if-state */
00481     if_state = (uint8_t) (if_state << 1);

00483         /* condition is true :) */

00490         /* skip if-block, jump to "else" or "fi" */

00492             LOG_DEBUG("following instruction will be skipped");
00493             sub_i = i_fetch();  // our i-fetch routine does the trick
00494             if (sub_i == I_UNKNOWN) {
00495                 return; // something went wrong, cancel
00496             } else if (sub_i == I_IF) {
00497                 ++depth;
00498             } else if (sub_i == I_FI) {
00499                 if (depth == 0) {
00500                     break;
00501                 } else {
00502                     --depth;
00503                 }
00504             }
00505         }
00506         if (sub_i == I_ELSE) {
