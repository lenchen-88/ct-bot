00145 void bot_ubasic_behaviour(Behaviour_t * data) {
00146     /* keine neue Zeile ausfuehren, falls nach letztem WAIT noch gewartet werden soll */
00147     if (wait_until <= TIMER_GET_TICKCOUNT_32) {
00148         LOG_DEBUG("neue Zeile von uBasic verarbeiten");
00149         ubasic_run();
00150         if (ubasic_finished()) {
00151             LOG_DEBUG("->-> Endeanforderung <-<-");
00152             return_from_behaviour(data);
00153             return;
00154         }
00155     } else {
00156         LOG_DEBUG("WAITing");
00157     }
00158 
00159     /* Speedvariable setzen */
00160     speedWishLeft = speedWishLeftBas;
00161     speedWishRight = speedWishRightBas;
00162 }
