00171 static int8_t access_field_lowres(position_t field, int8_t value, uint8_t set) {
00172     int16_t section_x, section_y, index_x, index_y;
00173 
00174     // Berechne in welcher Sektion sich der Punkt befindet
00175     section_x = field.x / MAP_SECTION_POINTS_LOWRES;
00176     section_y = field.y / MAP_SECTION_POINTS_LOWRES;
00177 
00178     if ((section_x >= MAP_SECTIONS_LOWRES)
00179             || (section_y >= MAP_SECTIONS_LOWRES)) {
00180         LOG_ERROR("Versuch ein Feld ausserhalb der Karte zu lesen!! x=%d y=%d\n", field.x, field.y);
00181         return 0;
00182     }
00183 
00184     // Berechne den Index innerhalb der Section
00185     index_x = field.x % MAP_SECTION_POINTS_LOWRES;
00186     index_y = field.y % MAP_SECTION_POINTS_LOWRES;
00187 
00188     if (set) { // Schreibzugriff
00189         // Eventuell existiert die Section noch nicht
00190         if (map_lowres[section_x][section_y] == NULL) {
00191             // Dann anlegen
00192             map_lowres[section_x][section_y] = malloc(
00193                     sizeof(map_section_lowres_t));
00194             if (map_lowres[section_x][section_y] == NULL) {
00195                 deactivateBehaviour(bot_calc_wave_behaviour);
00196                 return 0;
00197             }
00198             memset(map_lowres[section_x][section_y], 0,
00199                     sizeof(map_section_lowres_t));

00202         return value; // Schluss mit Werterueckgabe nach Schreibzugriff
00203     }
00204 
00205     // Eventuell existiert die Section noch nicht
00206     if (map_lowres[section_x][section_y] == NULL) {
00207         return 0;

00286         uint8_t end_reached, uint8_t * neighbour_found) {
00287     // nur im gueltigen Bereich
00288     if (map.x < 0 || map.y < 0 || map.x < min_x || map.x > max_x || map.y
00289             < min_y || map.y > max_y)
00290         return end_reached;
00291 
00292     // Mapwert auslesen aus uebergebener Koordinate des Nachbarn
00293     int8_t mapval = access_field_lowres(map, 0, 0);
00294 
00295     // Wellenwert wird nur auf noch initiale Felder eingetragen
00296     if (mapval == 0) { // nicht fuer Hindernis
00297         // in Lowres-Map Wellenwert auf die Koordinaten vermerken
00298         access_field_lowres(map, actual_wave, 1);
00299 
00300         // Nachbarpunkt als gueltig kennzeichnen
00301         *neighbour_found = True;
00302 
00303         // in FIFO-Queue aufnehmen zur weiteren Nachbarsuche
00304         if (!pos_store_queue(planning_pos_store, map)) {
00305             LOG_DEBUG(">> -Queue voll- %1d %1d bei Punkt: %1d %1d zu Welle: %1d", map.x, map.y, actual_wave);
00306         }
00307     } // nur initiale Felder
00308 
00309     // Ziel gefunden
00310     if (map.x == endkoord.x && map.y == endkoord.y) {
00311         LOG_DEBUG("Startkoord erreicht auf %1d %1d", map.x, map.y);
00312         return True;
00313     }
00314

00316     return end_reached;

00386     static position_t pos = { 0, 0 };
00387 
00388     // Zaehlervariablen zum Bestimmen der Nachbarn
00389     static int8_t i = 0;
00390     static int8_t j = 0;
00391 
00392     // Kennung ob eine gueltige Nachbarzelle gefunden werden konnte
00393     static uint8_t neighbour_found = 0;
00394 
00395     // Endekennung
00396     static uint8_t endreached = False; // Kennung gesetzt fuer Ziel gefunden; Terminierung der Schleife
00397 
00398     static uint8_t skip_count = 0; // Zaehlt die Anzahl der in der Queue uebersprungenen Punkte
00399 
00400     switch (wave_state) {
00401     // zuerst loeschen der Planungs-LowRes-Karte
00402     case START:
00403         LOG_DEBUG("Loeschen der Lowres-Karte");
00404         delete_lowres();
00405         planning_pos_store = pos_store_create_size(data, pos_store_data, QUEUE_SIZE); // Stack / Queue anlegen / und leeren
00406         wavecounter = 2; // geht ab Wert 2 los; d.h. Wert Wellenzentrum - Zielpunkt bekommt diesen Wert
00407         access_field_lowres(startwave, 2, 1);
00408 
00409         wave_state = SET_HAZARDS_TO_LOWRES;
00410         break;
00411 
00412     // Uebertragen der in der Highres-Karte vermerkten Hindernisse in die Lowres-Planungs-Karte
00413     case SET_HAZARDS_TO_LOWRES:
00414         LOG_DEBUG("Hindernisse eintragen");
00415         set_hazards(); // Hindernisse aus der Highres-Karte in die Planungs-Lowres-Karte eintragen
00416         wave_state = NEIGHBOURS_FROM_FIFO;
00417         break;
00418 
00419     // hier abarbeiten aller in der FIFO-Queue eingetragenen Koordinaten, d.h. fuer jede wird wieder der gueltige Nachbar in die Queue hinten eingefuegt bis Ende erreicht
00420     case NEIGHBOURS_FROM_FIFO:
00421         wavecounter++; // die ersten Nachbarn haben den Startwellenwert (2) + 1
00422         endreached = False; // Abbruchbedingung der Schleife init.
00423 
00424         pos_store_queue(planning_pos_store, startwave); // Startpunkt der Welle in die Queue einfuegen, damit gehts los
00425 
00426         LOG_DEBUG("Wellenstart bei %1d %1d Welle %1d", startwave.x, startwave.y, wavecounter);
00427 
00428         pos_store_queue(planning_pos_store, (position_t) {999, 999}); // nach Abarbeiten des Wellenstartpunktes muss Wellenzaehler erhoeht werden
00429 
00430         // Solange FIFO-Queue durchgehen und Nachbarn wieder anfuegen bis Queue leer ist oder Ziel (Botstartpunkt) gefunden (1 Nachbar ist Zielfeld)
00431         while (!endreached && pos_store_dequeue(planning_pos_store, &pos)) {
00432             // bei Kennung 999 ist Welle abgearbeitet, d.h. zu allen Punkten der Welle die Nachbarn wieder hinten angefuegt und Wellenzaehler wird erhoeht
00433             // wenn die Nachbarn der neuen Welle beginnen wird Wellenzaehler erhoeht
00434             if (pos.x == 999 && pos.y == 999) {
00435                 // die letzte in Queue eingefuegte Koord wird sich gemerkt; wenn zu dieser bei naechster Welle alle Nachbarn
00436                 // bearbeitet sind, ist naechste Welle vorbei
00437                 wavecounter++;
00438 
00439                 // Erhoehung fuer Wellenzaehler vermerken wenn Endepunkt noch nicht gefunden ist und gueltiger Nachbarpunkt vorhanden war
00440                 if (!endreached && neighbour_found) {
00441                     pos_store_queue(planning_pos_store, (position_t) {999, 999});
00442                 }
00443 
00444                 // Kennung fuer gueltigen Nachbarn wieder ruecksetzen
00445                 neighbour_found = False;
00446             } else {

00448                 // nur 4 Nachbarn kann bei Pfadsuche auf befahrenem Gebiet zu keinem Pfad fuehren, wenn Bot bisher nur eine Fahrspur hat und schraeg faehrt, dann wuerde

00451                     for (i=-1; i<=1; i++) {

00455                             tmp.x += i;
00456                             tmp.y += j;
00457                             endreached = get_neighbour(tmp, (int8_t) wavecounter, endreached, &neighbour_found);
00458                         }
00459                     }
00460                 }
00461 
00462             }

00505         while (!endreached) {
00506             neighbour_found = False;
00507

00509             for (j=-1; (j<=1) && !endreached; j++) {
