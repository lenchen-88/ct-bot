00077 void bot_drive_stack_behaviour(Behaviour_t * data) {
00078     uint8_t get_pos;
00079 
00080     switch (drivestack_state) {
00081     case 0:
00082         // Koordinaten werden vom Stack geholt und angefahren; Ende nach nicht mehr erfolgreichem Pop
00083 
00084         /* wenn Fifo-Queue definiert, kann sowohl nach LIFO (Stack) oder FIFO (Queue) gefahren werden */
00085         if (go_fifo) {
00086             get_pos = pos_store_dequeue(pos_store, &pos);
00087         } else {
00088             get_pos = pos_store_pop(pos_store, &pos);
00089         }
00090 
00091         if (!get_pos) {
00092             drivestack_state = 1;
00093         } else {
00094             bot_goto_pos(data, pos.x, pos.y, 999);
00095         }
00096         break;
00097 
00098     default:
00099 //      pos_store_release(pos_store); // Speicher freigeben
00100         pos_store = NULL;
00101         return_from_behaviour(data);
00102         if (put_stack_active == 1) {
00103             /* put_stack_waypos wieder an */
00104             bot_save_waypos(NULL, optimized_push);
00105         }
00106         break;
00107     }

00246 void bot_save_waypos_behaviour(Behaviour_t * data) {
00247     static uint8_t skip_count = 0;
00248     switch (waypos_state) {
00249     case 0:
00250         pos_store = pos_store_new_size(data, STACK_SIZE);
00251         if (pos_store == NULL) {
00252             exit_behaviour(data, BEHAVIOUR_SUBFAIL);
00253             return;
00254         }
00255         set_pos_to_last(); // aktuelle Botposition wird zum ersten Stackeintrag und merken der Position
00256         bot_push_pos(last_pos.x, last_pos.y);
00257         waypos_state = 1;
00258         break;
00259     case 1: {
00260         // Botpos wird in den Stack geschrieben zum Startzeitpunkt des Verhaltens (Stack noch leer) oder
00261         // wenn die Entfernung zur zuletzt gemerkten Koordinate gewissen Abstand ueberschreitet oder
00262         // wenn Entfernung noch nicht erreicht ist aber ein gewisser Drehwinkel erreicht wurde
00263 
00264         uint8_t update = False;
00265         // Abstand zur letzten Position ueberschritten
00266         if (get_dist(last_pos.x, last_pos.y, x_pos, y_pos) > DIST_FOR_PUSH) {
00267             // kein Push notwendig bei gerader Fahrt voraus zum Sparen des Stack-Speicherplatzes
00268             if ((int16_t) heading != last_heading) {
00269                 update = True;
00270             }
00271 
00272             set_pos_to_last(); // Position jedenfalls merken
00273         }
00274 
00275         // bei Drehwinkelaenderung und Uberschreitung einer gewissen Groesse mit geringer Abstandsentfernung zum letzten Punkt kommt er in den Stack
00276         if (turned_angle(last_heading) > ANGLE_FOR_PUSH && get_dist(last_pos.x,
00277                 last_pos.y, x_pos, y_pos) > DIST_FOR_PUSH_TURN) {
00278             set_pos_to_last();
00279             update = True;
00280         }
00281 
00282         if (update) {
00283             LOG_DEBUG("Position (%d|%d) kommt in den Stack", x_pos, y_pos);
00284             position_t pos_0;
00285             pos_0.x = x_pos / 16;
00286             pos_0.y = y_pos / 16;
00287             if (optimized_push >= 1) {
00288                 /* Positionen entfernen, die auf einer Linie liegen */
00289                 position_t pos_1, pos_2;
00290                 if (pos_store_top(pos_store, &pos_1, 1) == True && pos_store_top(pos_store, &pos_2, 2) == True) {
00291                     pos_1.x /= 16;
00292                     pos_1.y /= 16;
00293                     if (abs(pos_1.x - pos_0.x) <= 1) {
00294                         pos_0.x = pos_1.x;
00295                     }
00296                     pos_2.x /= 16;
00297                     pos_2.y /= 16;
00298                     if (abs(pos_2.x - pos_1.x) <= 1) {
00299                         pos_2.x = pos_1.x;
00300                     }
00301                     int8_t m_1 = (int8_t) (pos_1.x == pos_2.x ? 100 : abs((pos_1.y - pos_2.y) / (pos_1.x - pos_2.x)));
00302                     int8_t m = (int8_t) (pos_0.x == pos_1.x ? 100 : abs((pos_0.y - pos_1.y) / (pos_0.x - pos_1.x)));
00303                     LOG_DEBUG(" pos_2=(%d|%d)", pos_2.x, pos_2.y);
00304                     LOG_DEBUG(" pos_1=(%d|%d)", pos_1.x, pos_1.y);
00305                     LOG_DEBUG(" pos  =(%d|%d)", pos_0.x, pos_0.y);
00306                     LOG_DEBUG("  m_1=%3d\tm=%3d", m_1, m);
00307                     LOG_DEBUG("  skip_count=%u", skip_count);
00308                     if (abs(m_1 - m) < MAX_GRADIENT) {
00309                         LOG_DEBUG("Neue Position auf einer Linie mit beiden Letzten");
00310                         if (skip_count < 3) {
00311                             LOG_DEBUG(" Verwerfe letzten Eintrag (%d|%d)", pos_1.x, pos_1.y);
00312                             pos_store_pop(pos_store, &pos_1);
00313                             skip_count++;
00314                         } else {
00315                             LOG_DEBUG(" Verwerfe Eintrag NICHT, skip_count=%u", skip_count);
00316                             skip_count = 0;
00317                         }
00318                     } else {
00319                         skip_count = 0;
00320                     }
00321                 }
00322                 if (optimized_push >= 2) {
00323                     /* Schleifen entfernen */
00324                     uint8_t i;
00325                     for (i=2; pos_store_top(pos_store, &pos_1, i); ++i) {
00326                         int32_t diff = get_dist(x_pos, y_pos, pos_1.x, pos_1.y);
00327                         LOG_DEBUG(" diff=%d", diff);
00328                         if (diff < MAX_POS_DIFF) {
00329                             LOG_DEBUG(" Position (%d|%d)@%u liegt in der Naehe", pos_1.x, pos_1.y, i);
00330                             uint8_t k;
00331                             for (k=i; k>1; --k) {
00332                                 pos_store_pop(pos_store, &pos_1); // die Positionen bis zur i-ten zurueck loeschen
00333                                 LOG_DEBUG(" Loesche (%d|%d) vom Stack", pos_1.x, pos_1.y);
00334                             }
00335                             i = 2;
00336                         }
00337                     }
00338                 }
00339             }
00340 
00341             bot_push_pos(x_pos, y_pos);
00342 #if defined PC && defined DEBUG
00343             pos_store_dump(pos_store);
00344 #endif
00345 
00346 #ifdef DEBUG
00347 #ifdef MAP_2_SIM_AVAILABLE
00348             command_write(CMD_MAP, SUB_MAP_CLEAR_LINES, 0, 0, 0);
00349             position_t pos_1, pos_2;
00350             pos_store_top(pos_store, &pos_0, 1);
00351             uint16_t i;
00352             for (i = 2; pos_store_top(pos_store, &pos_1, (uint8_t) i); ++i) {
00353                 map_draw_line_world(pos_0, pos_1, 0);
00354                 pos_0 = pos_1;
00355                 pos_1.x = pos_0.x - 16;
00356                 pos_1.y = pos_0.y;
00357                 pos_2.x = pos_0.x + 16;
00358                 pos_2.y = pos_0.y;
00359                 map_draw_line_world(pos_1, pos_2, 1);
00360                 pos_1.x = pos_0.x;
00361                 pos_1.y = pos_0.y - 16;
00362                 pos_2.x = pos_0.x;
00363                 pos_2.y = pos_0.y + 16;

00365             }
00366 #endif // MAP_2_SIM_AVAILABLE
00367 #endif // DEBUG

00370         break;
