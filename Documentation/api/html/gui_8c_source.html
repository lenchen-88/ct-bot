00182 static uint16_t gui_keypad_check(uint16_t rc5) {
00183     static uint8_t pressed = 0;
00184     static uint16_t last_rc5 = 0;
00185 
00186     if (keypad_callback == NULL) {
00187         /* keine Keypad-Eingabe aktiv */
00188         return rc5;
00189     }
00190 
00191     switch (rc5) {
00192     case RC5_CODE_DOT:
00193         keypad_mode = (uint8_t) ((keypad_mode + 1) & 1);
00194         return 0;
00195     case RC5_CODE_STOP:
00196         /* Abbruch */
00197         *keypad_buffer = 0;
00198         // no break
00199     case RC5_CODE_PLAY:
00200         /* fertig */
00201         keypad_callback(keypad_buffer);
00202         keypad_callback = NULL;
00203         memset(keypad_buffer, 0, 21);
00204         /* no break */
00205     case 0:
00206         /* keine neue Eingabe */
00207         return 0;
00208     }
00209 
00210     const uint8_t key = (uint8_t) (rc5 - RC5_CODE_0);
00211     if (key > 10) {
00212         /* ungueltige Eingabe */
00213         return 0;
00214     }
00215 
00216     if (!timer_ms_passed_32(&keypad_last_pressed, 1000) && rc5 == last_rc5) {
00217         /* dieselbe Taste wurde mehrfach gedrueckt */
00218         pressed++;
00219     } else {
00220         /* eins weiter schalten */
00221         keypad_result++;
00222         pressed = 0;
00223         LOG_DEBUG("neue Spalte:%u", keypad_result - keypad_buffer);
00224     }
00225 
00226     LOG_DEBUG("pressed=%u", pressed);
00227 
00228     if (pressed > 4) {
00229         /* Umlauf */
00230         pressed = 0;
00231     }
00232 
00233     LOG_DEBUG("key=%u", key);
00234     char data;
00235     while ((data = (char) ctbot_eeprom_read_byte(&gui_keypad_table[key][pressed])) == 0) {
00236         /* bei einigen Tasten ist nicht alles belegt */
00237         pressed = 0;
00238     }
00239 
00240     /* im numerischen Fall nur Ziffern auswaehlen */
00241     if (keypad_mode == 1) {
00242         uint8_t i = 4;
00243         while ((data = (char) ctbot_eeprom_read_byte(&gui_keypad_table[key][i])) == 0) {
00244             i--;
00245         }
00246         rc5 = 0; // naechsten Tastendruck als neue Taste registrieren
00247     }
00248 
00249     last_rc5 = rc5;
00250     keypad_last_pressed = TIMER_GET_TICKCOUNT_32;
00251     *keypad_result = data;
00252     LOG_DEBUG("data=%c", data);
00253     LOG_DEBUG("result=%s", keypad_buffer);
00254     return 0;
00255 }
