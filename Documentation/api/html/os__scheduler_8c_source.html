00093 void os_calc_utilization(void) {
00094     if (timer_ms_passed_16(&last_util_time, UTIL_LOG_TIME)) {
00095         os_stat_data_t data[OS_MAX_THREADS];
00096         uint8_t i;
00097 
00098         /* Daten aus den TCBs holen */
00099         uint16_t sum = 0;
00100         os_enterCS();
00101         for (i = 0; i < OS_MAX_THREADS; ++i) {
00102             data[i] = os_threads[i].statistics;
00103             sum += data[i].runtime;
00104             os_threads[i].statistics.runtime = 0;
00105             os_threads[i].statistics.missed_deadlines = 0;
00106         }
00107         os_exitCS();
00108 
00109         if (sum == 0) {
00110             return; // keine neuen Daten vorhanden
00111         }
00112 
00113         /* Auslastung in % berechnen */
00114         for (i = 0; i < OS_MAX_THREADS; ++i) {
00115             missed_deadlines[i] += data[i].missed_deadlines;
00116             uint32_t tmp = data[i].runtime;
00117             tmp *= 100;
00118             tmp /= sum;
00119             util_data[util_count][i] = (uint8_t) tmp;
00120         }
00121         util_count++;
00122         util_count &= MAX_UTIL_ENTRIES - 1;
00123     }

00191 void os_idle(void) {
00192     while(42) {
00193 #ifndef OS_KERNEL_LOG_AVAILABLE
00194         /* Idle-Counter wird inkrementiert und 3-mal gespeichert.
00195          * Durch die 2 Backups gibt es immer mindestens 2 Kopien,
00196          * die den korrekten Wert beinhalten. */
00197         uint32_t tmp = idle_counter[0];
00198         tmp++;
00199         idle_counter[0] = tmp;
00200         idle_counter[1] = tmp;
00201         idle_counter[2] = tmp;
00202         // 38 Zyklen pro Durchlauf => 2375 ns @ 16 MHz, 1900 ns @ 20 MHz
00203 #else
00204         if (kernel_log_fifo.count > 0) {
00205             kernel_log_t data;
00206             fifo_get_data(&kernel_log_fifo, &data, sizeof(data));
00207             LOG_RAW("%u\t%u>%u\n", data.time, data.from, data.to);

00217 void os_schedule(uint32_t tickcount) {
00218     /* Mutex abfragen / setzen => Scheduler ist nicht reentrant! */
00219     if (test_and_set((uint8_t *) &os_scheduling_allowed, 0) != 1) {
00220         os_scheduling_allowed = 2;
00221         return;
00222     }
00223 
00224     /* Naechsten lauffaehigen Thread mit hoechster Prioritaet suchen.
00225      * Das ist zwar in O(n), aber wir haben nur eine sehr beschraenkte Anzahl an Threads! */
00226     uint8_t i;
00227     Tcb_t * ptr = os_threads;
00228     // os_scheduling_allowed == 0, sonst waeren wir nicht hier
00229     for (i = os_scheduling_allowed; i < OS_MAX_THREADS; ++i, ++ptr) {
00230         if (ptr->stack != NULL) {
00231             /* Es existiert noch ein Thread in der Liste */
00232             if (ptr->wait_for->value == 0 && ptr->nextSchedule <= tickcount) {
00233                 /* Der Thread ist nicht blockiert */
00234                 if (ptr != os_thread_running) {
00235 #ifdef OS_KERNEL_LOG_AVAILABLE
00236                     if (kernel_log_on != 0) {
00237                         log_entry.time = (uint16_t) tickcount;
00238                         log_entry.from = (uint8_t) (os_thread_running - os_threads);
00239                         log_entry.to = (uint8_t) (ptr - os_threads);
00240                         fifo_put_data(&kernel_log_fifo, &log_entry, sizeof(log_entry));
00241                     }
00242 #endif // OS_KERNEL_LOG_AVAILABLE
00243 
00244 #ifdef MEASURE_UTILIZATION
00245                     os_thread_running->statistics.runtime += (uint16_t) ((uint16_t) tickcount - os_thread_running->lastSchedule);
00246 #endif
00247                     /* switch Thread */
00248                     ptr->lastSchedule = (uint16_t) tickcount;
00249                     //-- hier laeuft noch der alte Thread (SP zeigt auf os_thread_running's Stack) --//
00250                     os_switch_thread(os_thread_running, ptr);
00251                     //-- jetzt laeuft bereits der neue Thread (SP zeigt auf ptr's Stack) --//
00252                     // => return fuehrt den NEUEN Thread weiter aus -> (noch) KEIN Ruecksprung zum Caller
00253                     break;
00254                 } else {
00255                     /* aktiver Thread darf weiterlaufen */
00256                     // => return fuehrt den ALTEN Thread weiter aus -> Funktionsaufruf mit "normaler" Rueckkehr
00257                     break;
00258                 }
00259             }
00260         }
00261     }
00262 
00263     /* Mutex freigeben */
00264     os_scheduling_allowed = 1;
00265     /* Ruecksprung dorthin, wo der (neu) geschedulte Thread vor SEINER Unterbrechung war,
00266      * also nicht zwangsweise in die Funktion, die (direkt) vor dem Scheduler-Aufruf
00267      * ausgefuehrt wurde! */
00268 }
