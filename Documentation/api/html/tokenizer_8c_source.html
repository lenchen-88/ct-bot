00287 static int get_next_token(void) {
00288 #if !USE_PROGMEM && TOKENIZER_STANDARD
00289     struct keyword_token const *kt;
00290 #endif
00291     uint8_t i;
00292     int temp_token;
00293 #if TOKENIZER_STANDARD
00294     char k_temp[MAX_KEYWORD_LEN+1];
00295 #endif
00296 
00297     if(END_OF_PROG_TEXT) {
00298         return TOKENIZER_ENDOFINPUT;
00299     }
00300     if(isdigit(GET_CONTENT_PROG_PTR)) {
00301         last_value=0;
00302 
00303         #if UBASIC_HEX_BIN
00304         // Zeiger sichern, falls doch nicht Hex oder Bin
00305         PTR_TYPE temp_ptr = PROG_PTR;
00306         if (GET_CONTENT_PROG_PTR == '0') {
00307             INCR_PROG_PTR;
00308         }
00309         if (toupper(GET_CONTENT_PROG_PTR)=='X' || toupper(GET_CONTENT_PROG_PTR)=='B') {
00310             switch (toupper(GET_CONTENT_PROG_PTR)) {
00311                 // Hex-Format --> 0x12AB
00312                 case 'X':
00313                     INCR_PROG_PTR;
00314                     for(i = 0; i <= MAX_HEXLEN; ++i) {
00315                         if(!isxdigit(GET_CONTENT_PROG_PTR)) {
00316                             if(i > 0) return TOKENIZER_NUMBER; else return TOKENIZER_ERROR;
00317                         }
00318                         last_value = 16 * last_value + hex2int((char) toupper(GET_CONTENT_PROG_PTR));
00319                         INCR_PROG_PTR;
00320                     }
00321                     return TOKENIZER_ERROR;
00322                     break;
00323                 // Binaer-Format --> 0b1010
00324                 case 'B':
00325                     INCR_PROG_PTR;
00326                     for(i = 0; i <= MAX_BINLEN; ++i) {
00327                         if(!isbdigit(GET_CONTENT_PROG_PTR)) {
00328                             if(i > 0) return TOKENIZER_NUMBER; else return TOKENIZER_ERROR;
00329                         }
00330                         last_value = 2 * last_value + GET_CONTENT_PROG_PTR - '0';
00331                         INCR_PROG_PTR;
00332                     }
00333                     return TOKENIZER_ERROR;
00334                     break;
00335             }
00336         }
00337         // kein Hex oder Bin, also wieder an Ursprungsstelle...
00338         SET_PROG_PTR_ABSOLUT(temp_ptr);
00339         #endif
00340         // Dezimal-Format
00341         for(i = 0; i <= MAX_NUMLEN; ++i) {
00342             if(!isdigit(GET_CONTENT_PROG_PTR)) {
00343                 if(i > 0) {
00344                     return TOKENIZER_NUMBER;
00345                 } else {
00346                     return TOKENIZER_ERROR;
00347                 }
00348             }
00349             last_value = 10 * last_value + GET_CONTENT_PROG_PTR - '0';
00350             INCR_PROG_PTR;
00351         }
00352         return TOKENIZER_ERROR;
00353     } else if(GET_CONTENT_PROG_PTR == '"') {
00354                         INCR_PROG_PTR;
00355                         i=0;
00356 
00357                         while (GET_CONTENT_PROG_PTR != '"') {
00358                             last_string[i] = GET_CONTENT_PROG_PTR;
00359                             INCR_PROG_PTR;
00360                             i++;
00361                             // max. zulaessige Stringlaenge?
00362                             if (i >= MAX_STRINGLEN) return TOKENIZER_ERROR;
00363                         }
00364 
00365                         // String null-terminieren
00366                         last_string[i]=0;
00367                         INCR_PROG_PTR;
00368                         return TOKENIZER_STRING;
00369                 } else {
00370 #if TOKENIZER_STANDARD
00371 
00372                     // Textabschnitt aus Quelltext auf Zwischenvariable
00373                     for (i=0; i < MAX_KEYWORD_LEN; i++) {
00374                         k_temp[i] = GET_CONTENT_PROG_PTR;
00375                         INCR_PROG_PTR;
00376                     }
00377                     // PTR wieder an Ursprungstelle setzen
00378                     SET_PROG_PTR_ABSOLUT(PROG_PTR - i);
00379                     k_temp[i]=0;
00380         #if USE_PROGMEM
00381                     for (i=0; i < sizeof(keywords)/sizeof(keywords[0])-1; i++) {
00382                         if (strncasecmp_P(k_temp, keywords[i].keyword, strlen_P(keywords[i].keyword)) == 0) {
00383                             SET_PROG_PTR_ABSOLUT(PROG_PTR + strlen_P(keywords[i].keyword));
00384                             return pgm_read_word(&keywords[i].token);
00385                         }
00386                     }
00387         #else
00388                     for(kt = keywords; kt->token != TOKENIZER_ERROR; ++kt) {
00389                         if(strncasecmp(k_temp, kt->keyword, strlen(kt->keyword)) == 0) {
00390                             SET_PROG_PTR_ABSOLUT(PROG_PTR + strlen(kt->keyword));
00391                             return kt->token;
00392                         }
00393                     }
00394         #endif
00395                 }
00396 
00397     temp_token = singlechar();
00398     if (temp_token) {
00399         INCR_PROG_PTR;
00400         return temp_token;
00401     }
00402 
00403 #else
00404     }
00405     // FastParser...
00406     temp_token = iFastParserGetKeyWord ();
00407     if (temp_token != -1) return temp_token;
00408 #endif
00409 
00410     if(toupper(GET_CONTENT_PROG_PTR) >= 'A' && toupper(GET_CONTENT_PROG_PTR) <= 'Z') {
00411         last_var_num=toupper(GET_CONTENT_PROG_PTR)-'A';
00412         INCR_PROG_PTR;
00413 #if UBASIC_STRING
00414         if (GET_CONTENT_PROG_PTR == '$') {
00415             INCR_PROG_PTR;
00416             return TOKENIZER_STRINGVAR;
00417         } else
00418 #endif
00419         return TOKENIZER_VARIABLE;
00420     }
00421     return TOKENIZER_ERROR;
00422 }

00511 int iFastParserGetKeyWord (void)  {
00512   char            ucCh;                 // Zeichen im Basic-Quell-Text
00513   unsigned char   ucRC,                 // "Referenz"-Zeichen (zu vergleichendes Zeichen im Baum)
00514                   ucS,                  // Niedrigstes erstes ASCII-Zeichen im Baum
00515                   ucE;                  // H√∂chstes erstes ASCII-Zeichen im Baum
00516   unsigned char  *pucPT = aucAVRBasic,  // Arbeitszeiger (mit Zeiger auf "Gesamtstruktur")
00517                  *pucTable;             // Zeiger auf Anfang des Baumes (nach Einsprungtabelle)
00518 
00519 #ifndef SFP_USES_16BIT

00521 #else

00523 #endif
00524 
00525   PTR_TYPE ptr_bak = PROG_PTR;          // Position (vor der Suche)
00526 
00527   // Aktuelles Zeichen aus dem Text holen
00528   ucCh      = GET_CONTENT_PROG_PTR;
00529 
00530   // Niedrigstes und h√∂chstes erstes Zeichen holen
00531   // (L√§nge der Einsprungtabelle)
00532 #if USE_PROGMEM                         // Tabelle liegt im Flash
00533     ucS = pgm_read_byte (&(pucPT [0]));
00534     ucE = pgm_read_byte (&(pucPT [1]));
00535 #else                                   // Tabelle liegt im RAM
00536     ucS = pucPT [0];
00537     ucE = pucPT [1];
00538 #endif
00539 
00540   // Startadresse des Baums holen
00541 #ifndef SFP_USES_16BIT                        // 8 bit Addressierung
00542     pucTable  = &(pucPT [2 + ucE - ucS]);
00543 #else                                         // 16 bit Addressierung
00544   pucTable  = &(pucPT [2 + (((unsigned short) (ucE - ucS)) << 1)]);
00545 #endif
00546 
00547    // Uppercase
00548     if ((ucCh >= 'A') && (ucCh <= 'Z')) {
00549         ucCh ^= 0x20;
00550     }
00551 
00552   // passt das erste Zeichen nicht ? dann Adios ..
00553     if ((ucCh < ucS) || (ucCh > ucE)) {
00554         SET_PROG_PTR_ABSOLUT (ptr_bak);
00555         return -1;
00556     }
00557 
00558   // Ja ..
00559 
00560 
00561   // ... dann Einsprung-Punkt in Baum holen
00562   ucCh = (char) (ucCh - ucS);
00563 #ifndef SFP_USES_16BIT                        // 8 bit Addressierung
00564   #if USE_PROGMEM                             // Tabelle liegt im Flash
00565     uiN   = pgm_read_byte (&(pucPT [2 + ucCh]));
00566   #else                                       // Tabelle liegt im RAM
00567     uiN   = pucPT [2 + ucCh];
00568     #endif
00569 #else                                         // 16 bit Addressierung
00570   #if USE_PROGMEM                             // Tabelle liegt im Flash
00571   uiN    = (unsigned short) (pgm_read_byte (&(pucPT [2 + (ucCh * 2)])) << 0);
00572   uiN   |= (unsigned short) (pgm_read_byte (&(pucPT [3 + (ucCh * 2)])) << 8);
00573   #else                                      // Tabelle liegt im RAM
00574   uiN    = (unsigned short) (&(pucPT [2 + (ucCh * 2)]) << 0);
00575   uiN   |= (unsigned short) (&(pucPT [3 + (ucCh * 2)]) << 8);
00576     #endif
00577 #endif
00578 
00579   // keinen Einsprung-Punkt ? dann Tsch√∂√∂ ..
00580 #ifndef SFP_USES_16BIT                        // 8 bit Addressierung
00581     if (uiN == 0xff) {
00582         SET_PROG_PTR_ABSOLUT (ptr_bak);
00583         return -1;
00584     }
00585 #else                                         // 16 bit Addressierung
00586   if (uiN == 0xffff) {
00587     SET_PROG_PTR_ABSOLUT (ptr_bak);
00588     return -1;
00589   }
