00156 int8_t command_read(void) {
00157     int8_t bytesRcvd;
00158     int8_t start = 0; // Start des Kommandos
00159     int8_t i;
00160     command_t * command; // Pointer zum Casten der empfangegen Daten
00161     uint8_t buffer[RCVBUFSIZE]; // Puffer
00162 #ifdef PC
00163 #if BYTE_ORDER == BIG_ENDIAN
00164     uint16_t store; // Puffer fuer die Endian-Konvertierung
00165 #endif
00166 #endif // PC
00167 
00168     uint16_t old_ticks; // alte Systemzeit
00169     buffer[0] = 0; // Sicherheitshalber mit sauberem Puffer anfangen
00170 
00171     /* Daten holen, maximal soviele, wie ein Kommando lang ist */
00172     bytesRcvd = (int8_t) low_read(buffer, sizeof(command_t));
00173 
00174 #ifdef DEBUG_COMMAND_NOISY
00175     LOG_DEBUG("%d read", bytesRcvd);
00176     LOG_DEBUG("%x %x %x", buffer[0], buffer[1], buffer[2]);
00177 #endif
00178     /* Suche nach dem Beginn des Frames */
00179     while ((start < bytesRcvd) && (buffer[start] != CMD_STARTCODE)) {
00180         LOG_DEBUG("falscher Startcode");
00181         start++;
00182     }
00183 
00184     /* Wenn keine STARTCODE gefunden ==> Daten verwerfen */
00185     if (buffer[start] != CMD_STARTCODE) {
00186         LOG_DEBUG("kein Startcode");
00187         return -1;
00188     }
00189 
00190 #ifdef DEBUG_COMMAND_NOISY
00191     LOG_DEBUG("Start @%d", start);
00192 #endif
00193 
00194     // haben wir noch genug Platz im Puffer, um das Packet ferig zu lesen?
00195     if ((RCVBUFSIZE - (uint8_t) (start)) < sizeof(command_t)) {
00196         LOG_DEBUG("not enough space");
00197         return -1; // nein? ==> verwerfen
00198     }
00199 
00200     i = (int8_t) ((int8_t) sizeof(command_t) - (bytesRcvd - start));
00201 
00202     if (i > 0) { // Fehlen noch Daten ?
00203         LOG_DEBUG("Start @ %d es fehlen %d Bytes ", start, i);
00204         old_ticks = TIMER_GET_TICKCOUNT_16; // Systemzeit erfassen
00205 
00206         /* So lange Daten lesen, bis das Packet vollstaendig ist, oder der Timeout zuschlaegt */
00207         while (i > 0) {
00208             if (timer_ms_passed_16(&old_ticks, COMMAND_TIMEOUT)) {
00209                 /* Timeout ueberschritten */
00210                 LOG_DEBUG("Timeout beim Nachlesen");
00211                 return -1; // ==> Abbruch
00212             }
00213             LOG_DEBUG("%d Bytes missing", i);
00214             i = (int8_t) low_read(buffer + bytesRcvd, (uint8_t) i);
00215             LOG_DEBUG("%d read", i);
00216             bytesRcvd = (int8_t) (bytesRcvd + i);
00217             i = (int8_t) ((int8_t) sizeof(command_t) - (bytesRcvd - start));
00218         }
00219     }
00220 
00221 #ifdef DEBUG_COMMAND_NOISY
00222     LOG_DEBUG("%d/%d read/start", bytesRcvd, start);
00223     LOG_DEBUG("%x %x %x", buffer[start], buffer[start+1], buffer[start+2]);
00224 #endif
00225 
00226     command = (command_t *) (buffer + start); // Cast in command_t
00227 
00228 #ifdef DEBUG_COMMAND_NOISY
00229     LOG_DEBUG("start: %c ", command->startCode);
00230     //  command_display(command);
00231 #endif
00232 
00233     // validate (Startcode ist bereits ok, sonst waeren wir nicht hier)
00234     if (command->CRC == CMD_STOPCODE) {
00235 #ifdef DEBUG_COMMAND_NOISY
00236         LOG_DEBUG("Command is valid");
00237 #endif
00238 
00239 #ifdef CHECK_CMD_ADDRESS
00240         /* Ist das Paket ueberhaupt fuer uns? */
00241         if ((command->to != CMD_BROADCAST)
00242                 && (command->to != get_bot_address())
00243                 && (command->request.command != CMD_WELCOME)) {
00244             LOG_DEBUG("Fehler: Paket To= %d statt %u", command->to, get_bot_address());
00245 #ifdef LOG_AVAILABLE
00246             command_display(command);
00247 #endif
00248             return -1;
00249         }
00250 #endif // CHECK_CMD_ADDRESS
00251 
00252         // Transfer
00253         memcpy(&received_command, buffer + start, sizeof(command_t));
00254 #ifdef PC
00255 #if BYTE_ORDER == BIG_ENDIAN
00256         /* Umwandeln der 16 bit Werte in Big Endian */
00257         store = received_command.data_l;
00258         received_command.data_l = store << 8;
00259         received_command.data_l |= (store >> 8) & 0xff;
00260 
00261         store = received_command.data_r;
00262         received_command.data_r = store << 8;
00263         received_command.data_r |= (store >> 8) & 0xff;
00264 #endif // BYTE_ORDER == BIG_ENDIAN
00265 #endif // PC
00266         return 0;
00267     } else { // Command not valid
00268         LOG_ERROR("Invalid Command:");
00269         LOG_DEBUG("%x %x %x", command->startCode, command->request.command, command->CRC);
00270         return -1;
00271     }
00272 }
