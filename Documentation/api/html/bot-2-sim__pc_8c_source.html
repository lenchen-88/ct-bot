00092 int8_t receive_until_Frame(uint8_t frame) {
00093     int8_t result;
00094     for (;;) {

00097             /* Fehler werden in command_read() ausgegeben
00098              * -1 kommt auch, wenn das Paket nicht fuer unsere Adresse war */

00101             command_evaluate();
00102         }
00103 
00104         if (received_command.request.command == frame) {
00105             return 0;
00106         }
00107     }
