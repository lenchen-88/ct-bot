00438 void bot_2_bot_handle_payload_ack(command_t * cmd) {
00439     if (bot_2_bot_data == NULL) {
00440         LOG_DEBUG("ACK von Bot %u empfangen, aber gar kein Transfer aktiv!", cmd->from);
00441         return;
00442     }
00443     switch (cmd->data_r) {
00444     case 0: {
00445             /* Der andere Bot hat unsere Anfrage akzeptiert */
00446             uint8_t window_size = (uint8_t) cmd->data_l;
00447             int16_t to_send = bot_2_bot_payload_size;
00448             LOG_DEBUG(" ACK von Bot %u, habe noch %d Bytes zu senden", cmd->from, to_send);
00449             LOG_DEBUG("  Bot %u hat window_size=%d festgelegt", cmd->from, window_size);
00450             to_send -= window_size;
00451             int16_t last_packet = 0;
00452             if (to_send < 0) {
00453                 /* Rest */
00454                 window_size = (uint8_t) (window_size + to_send);
00455                 to_send = 0;
00456                 last_packet = 1;
00457             }
00458             bot_2_bot_payload_size = to_send;
00459             LOG_DEBUG(" Sende %d Bytes zu Bot %u", window_size, cmd->from);
00460             command_write_rawdata_to(BOT_CMD_PAYLOAD, 0, cmd->from, last_packet, 0,
00461                     window_size, bot_2_bot_data);
00462 #ifdef PC
00463 
00464             command_write(CMD_DONE, SUB_CMD_NORM, simultime, 0, 0);
00465 #endif
00466             bot_2_bot_data += window_size;
00467             break;
00468         }
00469 
00470     case 1:
00471         /* Abbruch */
00472         LOG_DEBUG(" Bot %u hat Anfrage abgelehnt", cmd->from);
