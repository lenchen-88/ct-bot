00083 void bot_turn_behaviour(Behaviour_t * data) {
00084     static int16_t diff; // aktuelle Winkeldifferenz gegenueber Zielwinkel
00085     static uint16_t * p_err;
00086 
00087     /* Differenz zum Zielwinkel berechnen (in Drehrichtung gesehen) */
00088     if (turn_direction < 0) { // Uhrzeigersinn
00089         if (heading_10_int > old_heading && old_heading < 50) {
00090             target += 3600; // es gab einen Ueberlauf von heading
00091 #ifdef BEHAVIOUR_TURN_TEST_AVAILABLE
00092             target_fl += 3600.0f;
00093 #endif
00094 //          LOG_DEBUG("Ueberlauf");
00095 //          LOG_DEBUG("head=%d, old=%d", heading_10_int, old_heading);
00096         }
00097         diff = heading_10_int - target + 3600;
00098     } else {
00099         if (heading_10_int < old_heading && old_heading > 3550) {
00100             target -= 3600; // es gab einen Ueberlauf von heading
00101 #ifdef BEHAVIOUR_TURN_TEST_AVAILABLE
00102             target_fl -= 3600.0f;
00103 #endif
00104 //          LOG_DEBUG("Ueberlauf");
00105 //          LOG_DEBUG("head=%d, old=%d", heading_10_int, old_heading);
00106         }
00107         diff = target - (heading_10_int + 3600);
00108     }
00109     old_heading = heading_10_int;
00110 
00111     switch (state) {
00112         case STATE_RUNNNING: {
00113             int16_t last_speed = (abs(v_enc_left) + abs(v_enc_right)) / 2;
00114             if (last_speed < ERR_SPEED_1) {
00115                 p_err = &turn_err[0];
00116             } else if (last_speed < ERR_SPEED_2) {
00117                 p_err = &turn_err[1];
00118             } else if (last_speed < ERR_SPEED_3) {
00119                 p_err = &turn_err[2];
00120             } else if (last_speed < ERR_SPEED_4) {
00121                 p_err = &turn_err[3];
00122             } else if (last_speed < ERR_SPEED_5) {
00123                 p_err = &turn_err[4];
00124             } else {
00125                 p_err = &turn_err[5];
00126             }
00127             uint16_t err_ee = ctbot_eeprom_read_word(p_err);
00128             if (err_ee > 1800) {
00129                 err_ee = 0;
00130                 ctbot_eeprom_update_word(p_err, 0);
00131             }
00132             int16_t err = (int16_t) err_ee;
00133 
00134             if (diff > err) {
00135                 /* Bot drehen, solange Zielwinkel noch nicht erreicht ist */
00136                 int16_t new_speed = 0;
00137                 int16_t diff_9 = diff - 900;
00138                 if (diff_9 > 0) {
00139                     float x = diff_9 < 900 ? diff_9 / (float) (360.0 / M_PI * 10.0) : (float) (M_PI / 2.0); // (0; pi/2]
00140                     new_speed = (int16_t) (sinf(x) * (float) d_max_speed); // [0; maxspeed - minspeed]
00141                 }
00142                 new_speed = new_speed + min_speed; // [min_speed; maxspeed]
00143 
00144                 speedWishRight = turn_direction < 0 ? -new_speed : new_speed;
00145                 speedWishLeft  = -speedWishRight;
00146             } else {
00147                 /* Drehwinkel erreicht, Stopp und Abwarten, bis Nachlauf beendet */
00148                 speedWishLeft  = BOT_SPEED_STOP;
00149                 speedWishRight = BOT_SPEED_STOP;
00150                 state = STATE_WAIT_FOR_STOP;
00151                 LOG_DEBUG("last_speed=%u", last_speed);
00152             }
00153             break;
00154         }
00155 
00156         case STATE_WAIT_FOR_STOP: {
00157             if (v_enc_left == 0 && v_enc_right == 0) {
00158                 state = STATE_END;
00159             }
00160             break;
00161         }
00162 
00163         case STATE_END: {
00164             LOG_DEBUG("done, heading=%d", heading_int);
00165             LOG_DEBUG("target=%d", target / 10 - 360);
00166 
00167             /* Nachlauf beendet, jetzt Drehfehler aktualisieren */
00168             LOG_DEBUG("Fehler: %d.%u Grad", diff / 10, abs(diff - diff / 10 * 10));
00169 #ifdef BEHAVIOUR_TURN_TEST_AVAILABLE
00170             turn_last_err = fabs(target_fl / 10.0f - 360.0f - heading);
00171 #endif
00172             int16_t err = (int16_t) ctbot_eeprom_read_word(p_err);
00173             LOG_DEBUG("old err=%d", err);
00174             int16_t new_err = err - diff;
00175             if (new_err < 0) {
00176                 new_err = 0;
00177             }
00178             LOG_DEBUG("new_err=%d", new_err);
00179             uint16_t err_update = (uint16_t) ((err + new_err) / 2);
00180             LOG_DEBUG("updating err to %u", err_update);
00181             ctbot_eeprom_update_word(p_err, err_update);
00182 
00183             /* fertig, Verhalten beenden */
00184             return_from_behaviour(data);
00185             break;
00186         }
00187     }

00199 Behaviour_t * bot_turn_speed(Behaviour_t * caller, int16_t degrees, int16_t minspeed, int16_t maxspeed) {
00200     /* Parameter begrenzen */
00201     while (degrees >  360) degrees -= 360;
00202     while (degrees < -360) degrees += 360;
00203     int16_t tmp = maxspeed >= BOT_SPEED_FAST ? BOT_SPEED_FAST : maxspeed;
00204     d_max_speed = tmp <= BOT_SPEED_MIN ? BOT_SPEED_MIN : tmp;
00205     tmp = minspeed > d_max_speed ? d_max_speed : minspeed;
00206     min_speed = tmp < BOT_SPEED_MIN ? BOT_SPEED_MIN : tmp;
00207     d_max_speed -= min_speed;
00208 
00209     LOG_DEBUG("degrees=%d, min_speed=%d, d_max_speed=%d", degrees, min_speed, d_max_speed);
00210 
00211     /* Zielwinkel berechnen */
00212 #ifdef BEHAVIOUR_TURN_TEST_AVAILABLE
00213     target_fl = heading * 10.0f + degrees * 10.0f + 3600.0f;
00214 #endif
00215     target = heading_10_int + degrees * 10 + 3600;
00216     old_heading = heading_10_int;
00217 
00218     /* Drehrichtung ermitteln */
00219     turn_direction = (int8_t) (degrees < 0 ? -1 : 0);
00220 
00221     /* Verhalten aktiv schalten */
00222     state = STATE_RUNNNING;
00223     return switch_to_behaviour(caller, bot_turn_behaviour, BEHAVIOUR_OVERRIDE);
