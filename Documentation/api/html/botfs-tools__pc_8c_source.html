00572         botfs_file_descr_t file;
00573         char filename[BOTFS_MAX_FILENAME + 1];
00574         strncpy(filename, name, BOTFS_MAX_FILENAME);
00575         if (botfs_open(filename, &file, BOTFS_MODE_r, buffer) != 0) {
00576             printf("Fehler, konnte Datei \"%s\" nicht oeffnen!\n", filename);
00577             return;

00579         const uint16_t size = file.end - file.start + 1;
00580         if (size >= 2) {
00581             printf("Eintrag %3u:%15s\tstart:0x%04x\t\tend:0x%04x\n\t\t\t\tsize: %05u KByte\t", i, filename,

00583         } else {
00584             printf("Eintrag %3u:%15s\tstart:0x%04x\t\tend:0x%04x\n\t\t\t\tsize: %05u  Byte\t", i, filename,
00585                 file.start, file.end, size * BOTFS_BLOCK_SIZE);
00586         }
00587         const uint32_t used_size = file.used.start != UINT16_MAX ? (file.used.end - file.start) * BOTFS_BLOCK_SIZE + file.used.bytes_last_block : 0;
00588         if (used_size < 1024) {
00589             printf("used-size: %u Byte\n", used_size);

00591             printf("used-size: %u KByte\n", used_size / 1024);
