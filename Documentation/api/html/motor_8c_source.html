00099 void speed_control(uint8_t dev, int16_t * actVar, uint16_t * encTime, uint8_t i_time, uint8_t enc) {

00101     static int16_t lastErr[2] = {0,0};
00102     static int16_t last2Err[2] = {0,0};
00103     static uint8_t orignalTargetRate[2] = {0,0};
00104 
00105     if (encoderTargetRate[dev] == 0) {
00106         /* Fuer Stopp einfach alles nullsetzen */
00107         *actVar = 0;
00108         lastErr[dev] = 0;
00109         last2Err[dev] = 0;
00110     } else {
00111         /* Zeitdifferenz zwischen aktueller und ([vor- | 4.-])letzter Encoderflanke berechnen [176 us] */
00112         register uint8_t * p_time = (uint8_t *) encTime;
00113         uint16_t ticks_to_speed;
00114         uint16_t dt = *(uint16_t *) (p_time + i_time);  // aktueller Timestamp
00115         int8_t enc_correct = 0;
00116 
00117         /* Beim ersten Aufruf mit neuem Speed beruecksichtigen wir die Beschleunigung */
00118         if (start_signal[dev] == PID_START_DELAY) {
00119             orignalTargetRate[dev] = encoderTargetRate[dev];    // Zielgeschwindigkeit merken
00120             encoderTargetRate[dev] = BOT_SPEED_SLOW / 2;    // laaaangsam anfahren
00121 
00122 #ifdef SPEED_LOG_AVAILABLE
00123             /* Daten loggen */
00124             register uint8_t index = slog_i[dev];
00125             if (index < 24) {
00126                 slog->data[dev][index].encRate = 1; // Regelgroesse
00127                 slog->data[dev][index].err = encoderTargetRate[dev]; // Regeldifferenz
00128                 slog->data[dev][index].pwm = *actVar; // Stellgroesse
00129                 slog->data[dev][index].targetRate = encoderTargetRate[dev]; // Fuehrungsgroesse
00130                 slog->data[dev][index++].time = tickCount.u32; // Timestamp
00131                 slog_i[dev] = (uint8_t) (index > 24 ? 0 : index); // Z/25Z
00132             }
00133 #endif // SPEED_LOG_AVAILABLE
00134 
00135         } else { // 1. Aufruf => es gibt noch keinen korrekten Timestamp in der Vergangenheit => bis zum 2. Aufruf nix tun
00136             if (encoderTargetRate[dev] >= PID_SPEED_THRESHOLD) {
00137                 i_time = (uint8_t) (i_time - 4 * sizeof(encTime[0])); // Index 4.letzter Timestamp
00138                 ticks_to_speed = TICKS_TO_SPEED_2;
00139             } else if (encoderTargetRate[dev] < PID_SPEED_THRESHOLD / 2) {
00140                 i_time = (uint8_t) (i_time - sizeof(encTime[0])); // Index letzter Timestamp
00141                 ticks_to_speed = TICKS_TO_SPEED_0;
00142                 /* Regelgroesse korrigieren, wenn mit jeder Encoderflanke gerechnet wurde */
00143                 enc_correct = dev == 0 ? ENC_CORRECT_L : ENC_CORRECT_R;
00144                 if (enc == 1)  {
00145                     enc_correct = (int8_t) -enc_correct;
00146                 }
00147             } else {
00148                 i_time = (uint8_t) (i_time - 2 * sizeof(encTime[0])); // Index vorletzter Timestamp
00149                 ticks_to_speed = TICKS_TO_SPEED_1;
00150             }
00151             i_time = (uint8_t) (i_time & 0xf);                      // Index auf 4 Bit begrenzen
00152             dt -=  *(uint16_t *) (p_time + i_time); // gewaehlten Timestamp subtrahieren
00153 
00154             /* Bei Fahrt Regelgroesse berechnen */
00155             uint8_t encoderRate = (uint8_t) (ticks_to_speed / dt); // <dt> = [37; 800] -> <encoderRate> = [229; 10]
00156             if (encoderRate > 6) {
00157                 encoderRate = (uint8_t) (encoderRate + enc_correct);
00158             }
00159             /* Regeldifferenz berechnen */
00160             int16_t err = (encoderTargetRate[dev] - encoderRate);
00161             int16_t diff;
00162 #ifdef ADJUST_PID_PARAMS
00163             /* PID-Koeffizienten berechnen, falls PID-Parameter variabel */
00164             int16_t q0 = Kp + Kd / PID_Ta;
00165             int16_t q1 = -Kp - 2 * Kd / PID_Ta + Ki * PID_Ta;
00166             int16_t q2 = Kd / PID_Ta;
00167 
00168         /* Stellgroesse mit PID-Reglergleichung berechnen */
00169             diff = (q0 * err + q1 * lastErr[dev] + q2 * last2Err[dev]) >> PID_SHIFT;
00170 #else
00171             diff = (Q0 * err + Q1 * lastErr[dev] + Q2 * last2Err[dev]) >> PID_SHIFT;
00172 #endif // ADJUST_PID_PARAMS
00173             *actVar += diff;
00174 
00175             /* berechnete Stellgroesse auf zulaessige Werte begrenzen */
00176             if (*actVar > PWMMAX) *actVar = PWMMAX;
00177             else if (*actVar < PWMMIN) *actVar = PWMMIN;
00178 
00179             /* PWM-Lookup updaten */
00180             if (diff != 0 && encoderTargetRate[dev] == BOT_SPEED_SLOW/2 && start_signal[dev] == 0) {
00181                 uint8_t lastErrors = (uint8_t) (abs(last2Err[dev]) + abs(lastErr[dev]) + abs(err));
00182                 uint8_t turn = 0;
00183                 if (direction.left != direction.right) turn = 2;
00184                 if (pwm_values[dev + turn].rating >= lastErrors) {
00185                     pwm_values[dev + turn].pwm = (uint8_t) (*actVar >> 1);
00186                     pwm_values[dev + turn].rating = lastErrors;
00187                 }
00188             }
00189 
00190             /* Regeldifferenzen speichern */
00191             last2Err[dev] = lastErr[dev];
00192             lastErr[dev]  = err;
00193 
00194             /* Regelgroesse speichern, falls Displayausgabe gewuenscht */
00195 #ifdef DISPLAY_REGELUNG_AVAILABLE
00196             encoderRateInfo[dev] = encoderRate;
00197 #endif
00198 
00199 #ifdef SPEED_LOG_AVAILABLE
00200             /* Daten loggen */
00201             register uint8_t index = slog_i[dev];
00202             if (index < 24) {
00203                 slog->data[dev][index].encRate = encoderRate; // Regelgroesse
00204                 slog->data[dev][index].err = err; // Regeldifferenz
00205                 slog->data[dev][index].pwm = *actVar; // Stellgroesse
00206                 slog->data[dev][index].targetRate = encoderTargetRate[dev]; // Fuehrungsgroesse
00207                 slog->data[dev][index++].time = tickCount.u32; // Timestamp
00208                 slog_i[dev] = (uint8_t) (index > 24 ? 0 : index); // Z/25Z
00209             }
00210 #endif // SPEED_LOG_AVAILABLE
00211         }
00212 
00213         if (start_signal[dev] > 0) {
00214             start_signal[dev]--;
00215             /* langsam beschleunigen - eigentlich muesste man das sinusartig tun, aber das ist zu aufwendig
00216              * Verbesserungsmoeglichkeit: Faktoren bei START_DELAY optimieren (=> Sinus) */
00217             if (start_signal[dev] == (uint8_t) (PID_START_DELAY * 0.75f)) {
00218                 encoderTargetRate[dev] = (uint8_t) (encoderTargetRate[dev] + ((orignalTargetRate[dev] - BOT_SPEED_SLOW / 2) >> 2)); // +1/4
00219             } else if (start_signal[dev] == (uint8_t) (PID_START_DELAY * 0.5f)) {
00220                 encoderTargetRate[dev] = (uint8_t) (encoderTargetRate[dev] + ((orignalTargetRate[dev] - BOT_SPEED_SLOW / 2) >> 2)); // +2/4
00221             } else if (start_signal[dev] == (uint8_t) (PID_START_DELAY * 0.25f)) {
00222                 encoderTargetRate[dev] = (uint8_t) (encoderTargetRate[dev] + ((orignalTargetRate[dev] - BOT_SPEED_SLOW / 2) >> 2)); // +3/4
00223             } else if (start_signal[dev] == 0) {
00224                 encoderTargetRate[dev] = orignalTargetRate[dev];
00225             }
00226         }
00227     }
00228     /* PWM-Wert aktualisieren */
00229     motor_update(dev);
00230 }
