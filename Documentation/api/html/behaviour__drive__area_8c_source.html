00313 static uint8_t observe_get_startpoint(int8_t checkside, trackpoint_t * observe, uint8_t * behavstate) {
00314     position_t map_pos;
00315     int8_t mapval;
00316 
00317     if (map_locked() == 1) {
00318         return False;
00319     }
00320 
00321     // nur wenn bot seit letztem Zugriff gewisse Strecke gefahren ist erfolgt Mapzugriff; Check erfolgt seitenabhaengig
00322     uint8_t mapaccess = (uint8_t) (checkside == TRACKLEFT ? check_map_after_distance(&observe_lastpos.point1.x,
00323         &observe_lastpos.point1.y) : check_map_after_distance(&observe_lastpos.point2.x, &observe_lastpos.point2.y));
00324 
00325     // Schluss bei Endeanforderung oder noch nicht weit genug gefahren
00326     if (endrequest || ! mapaccess) {
00327         *behavstate = (uint8_t) (endrequest ? TRACK_END : 0);
00328         return False;
00329     }
00330 
00331     // hier seitenabhaengig auf die Spur nebenan sehen auf Hoehe Mittelpunkt und Mapwert checken
00332     mapval = getpoint_side_dist(BOT_DIAMETER - SIDEDIST_MINUS,
00333             -DISTSENSOR_POS_FW, checkside, &map_pos);
00334 
00335     // falls Punkt bereits als befahren gekennzeichnet ist oder Hinderniswert hat gehts weiter mit Startpunktsuche
00336     if (mapval >= MAP_TRACKVAL || mapval < 0) {
00337         return False;
00338     }
00339 
00340     // hier ist also der Trackpunkt frei, dann merken und zur Endpunktsuche im Observer
00341     // zuweisen der Initwerte in die richtigen Puntvars
00342     set_point_to_lastpoint(&observe->point1, map_pos.x, map_pos.y);
00343     set_point_to_lastpoint(&observe->point2, 0, 0);
00344 
00345     // Startpunkt gefunden
00346     return True;
00347 }

00364 static uint8_t observe_get_endpoint(int8_t checkside, uint8_t * behavstate,
00365         trackpoint_t * observer, position_t * lastpoint) {
00366     position_t map_pos;
00367     int8_t mapval = 0;
00368 
00369     if (map_locked() == 1) {
00370         return False;
00371     }
00372 
00373     // nur wenn bot seit letztem Zugriff gewisse Strecke gefahren ist erfolgt Mapzugriff
00374     uint8_t mapaccess = (uint8_t) (checkside == TRACKLEFT ? check_map_after_distance(&observe_lastpos.point1.x,
00375         &observe_lastpos.point1.y) : check_map_after_distance(&observe_lastpos.point2.x, &observe_lastpos.point2.y));
00376 
00377     // weitere Checks nur wenn gewisse Strecke gefahren oder Endeanforderung besteht
00378     if (! endrequest && ! mapaccess) {
00379         return False;  // Endepunktsuche ungueltig
00380     }
00381 
00382     // hier seitenabhaengig auf die jeweilige Spur sehen, Mapwert holen und die Koordinaten des Punktes der Nebenbahn
00383     mapval = getpoint_side_dist(BOT_DIAMETER - SIDEDIST_MINUS, 10, checkside, //etwas voraus
00384             &map_pos);
00385 
00386     // Endepunktsuche nicht weiter verfolgen falls noch kein Startpunkt vorliegt
00387     if ((observer->point1.x == 0 && observer->point1.y == 0)) {
00388         *behavstate = (uint8_t) (endrequest ? TRACK_END : 0); // weiter mit Startpunktsuche oder Ende
00389         return False; // Endepunktsuche ungueltig
00390     }
00391 
00392     // Hindernis oder schon befahren erkannt aber noch kein gueltiger Zwischenendpunkt ermittelt; Ende falls
00393     // Endeanforderung kam sonst weiter mit Startpunktsuche
00394     if ((mapval < 0 || mapval >= MAP_TRACKVAL) && &lastpoint->x == 0 && &lastpoint->y == 0) {
00395         set_point_to_lastpoint(&observer->point1, 0, 0);
00396         *behavstate = (uint8_t) (endrequest ? TRACK_END : 0); // weiter mit Startpunktsuche
00397         set_point_to_lastpoint(lastpoint, 0, 0);
00398         return False; // Endepunktsuche ungueltig
00399     }
00400 
00401     // Abstand zum Startpunkt muss gueltig sein sonst ist Schluss
00402     if (! dist_valid(observer->point1, map_pos)) {
00403         *behavstate = (uint8_t) (endrequest ? TRACK_END : 1); // bleibt in Endepunktsuche wenn nicht Endeanforderung kam
00404         if (endrequest) {
00405             LOG_DEBUG("Nebenpunkt zum Startpunkt dist nicht valid %1.0f",
00406                 sqrtf(get_dist(observer->point1.x, observer->point1.y, map_pos.x, map_pos.y)));
00407         }
00408         return False; // Endepunktsuche ungueltig
00409     }
00410 
00411     // hier wird nun unterschieden ob Nebenspur schon befahren ist oder sich dort Hindernis befindet; in diesem Fall wird sich
00412     // die Strecke als gueltig bis zu dem Punkt vor dem Hindernis auf dem Stack gemerkt; weiter dann mit Startpunktsuche
00413     if (mapval < 0 || mapval >= MAP_TRACKVAL) { //Punkt auf Nebenbahn hat Hinderniswahrscheinlichkeit oder ist schon befahren
00414 
00415         // bei Hindernis auf aktuellem Punkt wird der letzte gemerkte Nicht-Hindernispunkt auf den Stack gespeichert, der
00416         // vorher auf Gueltigkeit gecheckt wird, d.h. muss bestimmten Mindestabstand zum Startpunkt haben und kein Hindernis sein
00417         set_point_to_lastpoint(&observer->point2, lastpoint->x, lastpoint->y);
00418 
00419         // zuletzt gueltigen Endpunkt auf Stack sichern
00420         push_stack_pos_line(observer->point1, *lastpoint);
00421 
00422         if (lastpoint->x != 0 || lastpoint->y != 0) {
00423             LOG_DEBUG("Autopush wg. Hind oder befahren xy: %1d %1d mapval: %1d", lastpoint->x, lastpoint->y, mapval);
00424         }
00425 
00426         // letzten gemerkten Punkt wieder als ungueltig kennzeichnen
00427         set_point_to_lastpoint(lastpoint, 0, 0);
00428 
00429     } else {
00430         // Punkt auf Nebenbahn hat hier keine Hinderniswahrscheinlichkeit und ist noch nicht befahren
00431 
00432         // aktuellen gueltigen Zwischenpunkt der Nebenbahn merken; wird als Endepunkt genommen bei Hindernis auf Nebenbahn
00433         set_point_to_lastpoint(lastpoint, map_pos.x, map_pos.y);
00434 
00435         // falls sich bei diesem gueltigen Endpunkt herausstellt, dass der vorhin ermittelte Startpunkt nun Hinderniswert besitzt, so wird der Endpunkt zum Startpunkt
00436         // und weiter mit Endpunktsuche
00437         if (map_get_field(observer->point1) < 0) {
00438             set_point_to_lastpoint(&observer->point1,lastpoint->x, lastpoint->y); // Startpunkt ueberschreiben
00439             *behavstate = (uint8_t) (endrequest ? TRACK_END : 1); // bleibt in Endepunktsuche wenn nicht Endeanforderung kam
00440 
00441             set_point_to_lastpoint(lastpoint, 0, 0);
00442             set_point_to_lastpoint(&observer->point2, 0, 0);
00443             return True; // gueltiger Endepunkt ermittelt
00444         }
00445 
00446         // bei Nicht Ende geht Endesuche weiter;bei Endeanforderung ist dies der aktuelle Endpunkt fuer den Stack
00447         if (!endrequest) { // weiter nur falls kein Ende kommen soll sonst ist dies nicht der echte Endpunkt
00448             return False;
00449         }
00450 
00451         // gueltigen Endepunkt in Observervar vermerken
00452         set_point_to_lastpoint(&observer->point2, lastpoint->x, lastpoint->y);
00453 
00454         // Merken der Strecke bei Endanforderung auf dem Stack
00455         push_stack_pos_line(observer->point1,*lastpoint);
00456 
00457     }
00458 
00459     // weiter mit Startpunktsuche wenn nicht Ende gefordert
00460     *behavstate = (uint8_t) (endrequest ? TRACK_END : 0);
00461 
00462     // solange die Startpunktsuche keinen neuen Startpunkt ermittelt hat, ist dieser initial
00463     set_point_to_lastpoint(&observer->point1, 0, 0);
00464 
00465     return True; // Streckenpunkt ist hier gueltig
00466 }

00477 void bot_observe_left_behaviour(Behaviour_t * data) {
00478     // je zu beobachtende Seite gibt es einen codeidentischen Observer; hier die Seite festgelegt
00479     static int8_t CHECK_SIDE = TRACKLEFT;
00480 
00481     switch (observe1_state) {
00482     case OBSERVE_SEARCH_STARTPOINT:
00483         // Eingang fuer Startpunktsuche
00484         if (!observe_get_startpoint(CHECK_SIDE, &observe_trackleft, &observe1_state)) {
00485             break;
00486         }
00487 
00488         observe1_state = OBSERVE_SEARCH_ENDPOINT;
00489         break;
00490 
00491     case OBSERVE_SEARCH_ENDPOINT:
00492         // Anfangspunkt wurde bestimmt und ist hier auf der Suche nach dem Endpunkt; ein Endpunkt wird als Ende der Strecke
00493         // vom bereits vorhandenen Startpunkt erkannt, wenn der Botbereich neben dem Bot nicht mehr befahren werden kann; es
00494         // wird dann der zuletzt gemerkte befahrbare Punkt als Endpunkt genommen
00495         if (!observe_get_endpoint(CHECK_SIDE, &observe1_state,
00496                 &observe_trackleft, &lastpointleft)) {
00497             break;
00498         }
00499 
00500         break;
00501     default:
00502         if (observe_trackleft.point2.x == 0 && observe_trackleft.point2.y == 0) {
00503             set_point_to_lastpoint(&observe_trackleft.point1, 0, 0);
00504         }
00505 
00506         return_from_behaviour(data);
00507         break;
