00170         LOG_DEBUG("__heap_end = 0x%04x", __brkval);

00172         LOG_DEBUG("RAMEND = 0x%04x", RAMEND);

00174     }
00175 #endif // RC5_AVAILABLE
00176 #endif // LOG_AVAILABLE
00177     size_t data_size = (size_t) (&__bss_start - &__data_start);
00178     size_t bss_size = (size_t) (&__heap_start - &__bss_start);
00179     display_cursor(1, 1);
00180     display_printf("bss/data:%4u/%4u B", bss_size, data_size);
00181     display_cursor(2, 1);
