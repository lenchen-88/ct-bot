00050 void bot_gotoxy_behaviour(Behaviour_t * data) {
00051     /* aus aktueller Position und Ziel neuen Zielwinkel berechnen */
00052     int16_t xDiff = target_x - x_pos;
00053     int16_t yDiff = target_y - y_pos;
00054     
00055     if(xDiff*initialDiffX < 0 || yDiff*initialDiffY < 0) {  // Hier kann noch verbessert werden
00056         gotoState=REACHED_POS;          // z.B. Abfragen statt *-Operation
00057         speedWishLeft=BOT_SPEED_STOP;       // bzw. neue Drehung statt Stehenbleiben
00058         speedWishRight=BOT_SPEED_STOP;
00059     }
00060 
00061     switch(gotoState) {
00062         case INITIAL_TURN:
00063             gotoState=GOTO_LOOP;
00064             bot_turn(data, (int16_t) calc_angle_diff(xDiff, yDiff));
00065             break;
00066 
00067         case GOTO_LOOP:
00068             /* Position erreicht? */
00069             if (abs(xDiff)<10 || abs(yDiff)<10) {
00070                 gotoState=CORRECT_HEAD;
00071                 bot_turn(data, (int16_t) calc_angle_diff(xDiff, yDiff));
00072                 break;
00073             }
00074             speedWishLeft=BOT_SPEED_FOLLOW;
00075             speedWishRight=BOT_SPEED_FOLLOW;
00076             break;
00077 
00078         case CORRECT_HEAD:
00079             /* Position erreicht? */
00080             if (abs(xDiff) < 3 && abs(yDiff) < 3) {
00081                 gotoState=REACHED_POS;
00082                 speedWishLeft=BOT_SPEED_STOP;
00083                 speedWishRight=BOT_SPEED_STOP;
00084                 break;
00085             }
00086             speedWishLeft=BOT_SPEED_SLOW;
00087             speedWishRight=BOT_SPEED_SLOW;
00088             break;
00089 
00090         case REACHED_POS:
00091             gotoState=INITIAL_TURN;
00092             return_from_behaviour(data);
00093             break;
00094     }
00095 }
