00416 static void read_command_thread(void) {
00417     char input[255];
00418 
00419     while (42) {
00420         putc('>', stdout);
00421         if (fgets(input, sizeof(input) - 1, stdin) == NULL) {
00422             continue;
00423         }
00424         if (*input == '\n' || strncmp(input, "list", strlen("list")) == 0) {
00425             int i = 0;
00426             while (remotecall_beh_list[i].func != NULL) {
00427                 printf("%s(%s)\n", remotecall_beh_list[i].name, remotecall_beh_list[i].param_info);
00428                 ++i;
00429             }
00430             continue;
00431         } else if (strncmp(input, "cancel", strlen("cancel")) == 0) {
00432             puts("RemoteCall abgebrochen.");
00433             bot_remotecall_cancel();
00434             continue;
00435         }
00436 
00437         char * function = strtok(input, "(\n");
00438         if (function == NULL || *function == '\n') {
00439             continue;
00440         }
00441         char * param[REMOTE_CALL_MAX_PARAM] = {NULL};
00442         remote_call_data_t params[REMOTE_CALL_MAX_PARAM] = {{0}};
00443         int i;
00444         for (i = 0; i < REMOTE_CALL_MAX_PARAM; ++i) {
00445             param[i] = strtok(NULL, ",)");
00446             if (param[i] != NULL && *param[i] != '\n') {
00447                 params[i].s32 = atoi(param[i]);
00448             }
00449         }
00450 
00451 #ifdef DEBUG
00452         printf("function=\"%s\"\n", function);
00453 #endif // DEBUG
00454 
00455         for (i = 0; i < REMOTE_CALL_MAX_PARAM; ++i) {
00456             if (param[i] != NULL && *param[i] != '\n') {
00457 #ifdef DEBUG
00458                 printf("param[%d]=\"%s\"\n", i, param[i]);
00459                 printf("params[%d].s32=%d\tparams[%d].u32=0x%x\n", i, params[0].s32, i, params[0].u32);
00460 #endif // DEBUG
00461             }
00462         }
00463 
00464         int8_t result = bot_remotecall(NULL, function, params);
00465         if (result == 0) {
00466             printf("RemoteCall \"%s(%d,%d,%d)\" gestartet\n", function, params[0].s32, params[1].s32, params[2].s32);
00467         } else if (result == -1) {

00472             result = bot_remotecall(NULL, func, params);

00474                 printf("RemoteCall \"%s(%d,%d,%d)\" gestartet\n", func, params[0].s32, params[1].s32, params[2].s32);
