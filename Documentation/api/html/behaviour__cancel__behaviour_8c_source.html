00056 void bot_behaviour_cancel_behaviour(Behaviour_t * data) {
00057     uint8_t idle = True;
00058     uint8_t i;
00059     for (i = 0; i < sizeof(jobs) / sizeof(jobs[0]); ++i) {
00060         if (jobs[i].beh != NULL) {
00061             if ((jobs[i].beh->active != BEHAVIOUR_ACTIVE && jobs[i].beh->subResult != BEHAVIOUR_SUBRUNNING )
00062                 || jobs[i].cond == NULL) {
00063                 LOG_DEBUG("erledigten Auftrag gefunden, wird entfernt");
00064                 jobs[i].beh = NULL;
00065                 continue;
00066             }
00067 //          LOG_DEBUG("zu ueberwachendes Verhalten %u gefunden", jobs[i].beh->priority);
00068             idle = False;
00069             if (jobs[i].cond() != 0) {
00070                 /* Check-Funktion vorhanden und Abbruchbedingung erfuellt */
00071                 LOG_DEBUG("Abbruch des Verhaltens %u", jobs[i].beh->priority);
00072                 deactivate_called_behaviours(jobs[i].beh); // vom Zielverhalten aufgerufene Verhalten beenden
00073                 exit_behaviour(jobs[i].beh, BEHAVIOUR_SUBCANCEL); // Zielverhalten beenden, Caller reaktivieren
00074                 jobs[i].beh = NULL;
00075             }
00076         }
00077     }
00078 
00079     if (idle) {
00080         /* keine Auftraege mehr -> sich selbst deaktivieren */
00081         LOG_DEBUG("keine weiteren Auftraege");
00082         deactivate_behaviour(data);
00083     }
00084 }

00093 Behaviour_t * bot_add_behaviour_to_cancel(Behaviour_t * caller, Behaviour_t * behaviour, uint8_t (* check)(void)) {
00094     (void) caller;
00095     if (behaviour == NULL) {
00096         LOG_DEBUG("Fehler, kein gueltiges Verhalten uebergeben");
00097         return NULL;
00098     }
00099 
00100     LOG_DEBUG("cancel(0x%x, 0x%x (Prio %u), 0x%x)", caller, behaviour, behaviour->priority, check);
00101     uint8_t i;
00102     for (i = 0; i < sizeof(jobs) / sizeof(jobs[0]); ++i) {
00103         if (jobs[i].beh == NULL) {
00104             LOG_DEBUG(" i=%u noch frei", i);
00105             jobs[i].beh = behaviour;
00106             jobs[i].cond = check;
00107             break;
00108         }
00109     }
00110     if (i == sizeof(jobs) / sizeof(jobs[0])) {
00111         LOG_DEBUG("Kein Slot mehr frei, Abbruch");
